function updateUI(){
  [1,2,3].forEach(i=>{
    const el=document.getElementById('s'+i);
    const nm=document.getElementById('sn'+i);
    el.classList.remove('active','done');
    if(i<S.step){el.classList.add('done');nm.textContent='✓';}
    else if(i===S.step){el.classList.add('active');nm.textContent=i;}
    else nm.textContent=i;
  });

  const impFlt=document.getElementById('btn-imp-flt');
  const savSys=document.getElementById('btn-save-sys');
  const savRes=document.getElementById('btn-save-res');

  if(S.sysLoaded){
    savSys.classList.remove('tbtn-disabled');
    savSys.classList.add('ready');
  }else{
    savSys.classList.add('tbtn-disabled');
    savSys.classList.remove('ready');
  }

  if(S.systemSaved){
    impFlt.classList.remove('tbtn-disabled');
    impFlt.classList.add('ready');
  }else{
    impFlt.classList.add('tbtn-disabled');
    impFlt.classList.remove('ready');
  }

  if(S.simDone){
    savRes.classList.remove('tbtn-disabled');
    savRes.classList.add('ready');
  }else{
    savRes.classList.add('tbtn-disabled');
    savRes.classList.remove('ready');
  }

  let statusText='就绪 · 等待导入系统模型';
  if(S.sysLoaded && S.step===1){
    if(!S.modelNodes.length){
      statusText='系统建模中 · 请拖拽组件到画布';
    }else if(canInitializeSimulation()){
      statusText='系统建模中 · 当前链路已可运行，可直接初始化仿真';
    }else{
      statusText='系统建模中 · 单击模块后可在右侧保存设置';
    }
  }else if(S.step===2){
    statusText=hasSimulationFaults()
      ?'系统模型已保存 · 当前包含故障注入，可继续仿真分析'
      :'系统模型已保存 · 故障注入为可选，可直接初始化仿真';
  }else if(S.step===3){
    statusText=hasSimulationFaults()
      ?'故障配置完成 · 仿真分析'
      :'仿真分析阶段 · 当前为无故障基线仿真';
  }
  if(typeof SIM!=='undefined'){
    if(SIM.running&&!SIM.paused){
      statusText=`仿真运行中 · t = ${SIM.time.toFixed(2)}s / ${SIM.duration.toFixed(2)}s`;
    }else if(SIM.paused){
      statusText=`仿真已暂停 · t = ${SIM.time.toFixed(2)}s / ${SIM.duration.toFixed(2)}s`;
    }else if(SIM.hasSamples&&S.step===3){
      statusText=`仿真结果就绪 · ${SIM.datasetName} · 已采样 ${SIM.stepIndex} 步`;
    }
  }
  document.getElementById('stxt').textContent=statusText;
  const sd=document.getElementById('sdot');
  sd.className='sdot'+['','sdot-b','sdot-r','sdot-g'][S.step];
  document.getElementById('sblk').textContent=`组件: ${S.modelNodes.length} · 连线: ${S.modelEdges.length}`;
  document.getElementById('sflt').textContent=`故障模型: ${S.faultModels}`;
  updateSimbarEnabled();
}

function clamp(value,min,max){return Math.min(Math.max(value,min),max);}

function cloneDefaults(value){
  return JSON.parse(JSON.stringify(value));
}

const WORKBENCH_STORAGE_KEY='gz-workbench-system-model';

function getRuntimeListenerStore(){
  if(!window.__GZ_RUNTIME_LISTENERS__){
    window.__GZ_RUNTIME_LISTENERS__={};
  }
  return window.__GZ_RUNTIME_LISTENERS__;
}

function getWorkbenchSnapshotApi(){
  return window.__GZ_WORKBENCH_SNAPSHOT__||null;
}

function createSystemModelSnapshot(){
  const api=getWorkbenchSnapshotApi();
  const payload={
    modelNodes:S.modelNodes,
    modelEdges:S.modelEdges,
    nodeSeq:S.nodeSeq,
    edgeSeq:S.edgeSeq,
    activeLineType:S.activeLineType,
    faultedBlks:S.faultedBlks,
    importedFaultModels:S.importedFaultModels
  };
  return api?.createWorkbenchSnapshot?api.createWorkbenchSnapshot(payload):cloneDefaults(payload);
}

function restoreSystemModelSnapshot(snapshot){
  const api=getWorkbenchSnapshotApi();
  const restored=api?.restoreWorkbenchSnapshot?api.restoreWorkbenchSnapshot(snapshot):cloneDefaults(snapshot);
  S.modelNodes=restored.modelNodes||[];
  S.modelEdges=restored.modelEdges||[];
  S.nodeSeq=restored.nodeSeq||0;
  S.edgeSeq=restored.edgeSeq||0;
  S.activeLineType=restored.activeLineType||'normal';
  S.faultedBlks=restored.faultedBlks||[];
  S.importedFaultModels=restored.importedFaultModels||[];
  S.faultModels=S.importedFaultModels.length;
  S.selBlk=null;
  S.selEdge=null;
  clearPendingConnection();
  syncAllPythonInputConnectionState();
  renderModelNodes();
  renderPropertyPanel(null);
  setConnectionTool(S.activeLineType);
  syncConfigTargets('');
}

function escapeHtml(value){
  return String(value ?? '')
    .replace(/&/g,'&amp;')
    .replace(/</g,'&lt;')
    .replace(/>/g,'&gt;')
    .replace(/"/g,'&quot;')
    .replace(/'/g,'&#39;');
}

function clampInterfaceCount(value,[min,max]){
  const parsed=Number.parseInt(value,10);
  if(!Number.isFinite(parsed)){return min;}
  return clamp(parsed,min,max);
}

function normalizeSimulationInterfaceList(items,prefix,fallbackType,minCount=0){
  const base=Array.isArray(items)?items:[];
  const normalized=base
    .map((item,index)=>({
      name:String(item?.name||`${prefix} ${index+1}`).trim()||`${prefix} ${index+1}`,
      type:PORT_FORMATS.includes(item?.type)?item.type:(fallbackType||'标量')
    }))
    .filter(Boolean);
  while(normalized.length<minCount){
    normalized.push({
      name:`${prefix} ${normalized.length+1}`,
      type:fallbackType||'标量'
    });
  }
  return normalized;
}

function normalizeSimulationInterfaces(props){
  const inputFallback=PORT_FORMATS.includes(props?.inputFormat)?props.inputFormat:'标量';
  const outputFallback=PORT_FORMATS.includes(props?.outputFormat)?props.outputFormat:'标量';
  props.inputs=normalizeSimulationInterfaceList(props?.inputs,'输入',inputFallback,1);
  props.outputs=normalizeSimulationInterfaceList(props?.outputs,'输出',outputFallback,1);
  props.middleVars=normalizeSimulationInterfaceList(props?.middleVars,'中间量',outputFallback,0);
  if(!props.middleVars.length&&props?.inputFormat){
    props.middleVars=[{name:'中间量 1',type:outputFallback}];
  }
  if(!props.ffunctionNote){
    props.ffunctionNote='当前预留 Python 函数接口，可在后续导入函数定义。';
  }
  delete props.inputFormat;
  delete props.outputFormat;
  return props;
}

function getSimulationPortSets(props){
  normalizeSimulationInterfaces(props);
  return {
    inputs:props.inputs,
    outputs:props.outputs,
    middleVars:props.middleVars
  };
}

function getSimulationNodeSize(props){
  const groups=getSimulationPortSets(props);
  const leftCount=Math.max(groups.inputs.length,1);
  const rightCount=Math.max(groups.outputs.length,1);
  const topCount=groups.middleVars.length;
  const verticalDominant=Math.max(leftCount,rightCount);
  return {
    width:clamp(196+Math.max(0,topCount-2)*28,196,336),
    height:clamp(116+Math.max(0,verticalDominant-2)*22+(topCount?18:0),116,312)
  };
}

function applyNodeGeometry(node){
  if(!node){return;}
  if(node.type==='flow_block'&&isPythonBoundFlowBlock(node)){
    const size=getPythonBindingNodeSize(node);
    node.w=size.width;
    node.h=size.height;
  }else if(node.type==='simulation_block'){
    normalizeSimulationInterfaces(node.props);
    const size=getSimulationNodeSize(node.props);
    node.w=size.width;
    node.h=size.height;
  }else{
    const meta=COMPONENT_LIBRARY[node.type];
    if(meta){
      node.w=meta.width;
      node.h=meta.height;
    }
  }
  node.x=clamp(node.x,CANVAS_STAGE.nodePadding,CANVAS_STAGE.width-node.w-CANVAS_STAGE.nodePadding);
  node.y=clamp(node.y,CANVAS_STAGE.nodeMinY,CANVAS_STAGE.height-node.h-CANVAS_STAGE.nodePadding);
}

function pruneInvalidEdgesForNode(node){
  const ports=getNodePorts(node);
  const outputCount=ports.outputs.length;
  const inputCount=ports.inputs.length;
  const removed=S.modelEdges.filter(edge=>{
    if(edge.sourceNodeId===node.id&&(edge.sourcePortIndex||0)>=outputCount){return true;}
    if(edge.targetNodeId===node.id&&(edge.targetPortIndex||0)>=inputCount){return true;}
    return false;
  });
  if(!removed.length){return 0;}
  const removedIds=new Set(removed.map(edge=>edge.id));
  S.modelEdges=S.modelEdges.filter(edge=>!removedIds.has(edge.id));
  syncAllPythonInputConnectionState();
  if(S.selEdge&&removedIds.has(S.selEdge)){S.selEdge=null;}
  if(S.protocolInjectTargetId&&removedIds.has(S.protocolInjectTargetId)){
    S.protocolInjectTargetId='';
    S.protocolInjectSelectionMode='';
    S.protocolInjectSelectionId='';
  }
  return removed.length;
}

function getSimulationFlatOutputs(props){
  const groups=getSimulationPortSets(props);
  return [
    ...groups.outputs.map((item,index)=>({kind:'output',name:item.name,type:item.type,index,side:'right'})),
    ...groups.middleVars.map((item,index)=>({kind:'middle',name:item.name,type:item.type,index,side:'top'}))
  ];
}

function findNearestEdgeIdByClientPoint(clientX,clientY,threshold=14){
  const paths=[...document.querySelectorAll('.edge-path[data-edge-id]')];
  if(paths.length===0){return '';}
  const point=clientToStagePoint(clientX,clientY);
  const stageThreshold=threshold/Math.max(S.canvasScale,0.001);
  let nearestId='';
  let nearestDistance=Infinity;

  paths.forEach(path=>{
    const total=path.getTotalLength();
    const sampleStep=Math.max(8,total/18);
    for(let cursor=0;cursor<=total;cursor+=sampleStep){
      const pos=path.getPointAtLength(Math.min(cursor,total));
      const distance=Math.hypot(pos.x-point.x,pos.y-point.y);
      if(distance<nearestDistance){
        nearestDistance=distance;
        nearestId=path.dataset.edgeId||'';
      }
    }
    const last=path.getPointAtLength(total);
    const lastDistance=Math.hypot(last.x-point.x,last.y-point.y);
    if(lastDistance<nearestDistance){
      nearestDistance=lastDistance;
      nearestId=path.dataset.edgeId||'';
    }
  });

  return nearestDistance<=stageThreshold?nearestId:'';
}

function getNodePorts(node){
  if(!node){return {inputs:[],outputs:[]};}
  if(node.type==='flow_block'&&isPythonBoundFlowBlock(node)){
    const binding=ensureFlowNodePythonBinding(node);
    return {
      inputs:binding.portMapping.inputs.map((item,index)=>({
        key:`input-${index}`,
        label:item.displayName||item.varName||`input_${index}`,
        kind:'input',
        side:'left',
        meta:item
      })),
      outputs:[
        ...binding.portMapping.outputs.map((item,index)=>({
          key:`output-${index}`,
          label:item.displayName||item.varName||`output_${index}`,
          kind:'output',
          side:'right',
          meta:item
        })),
        ...binding.portMapping.middleVars.map((item,index)=>({
          key:`middle-${index}`,
          label:item.displayName||item.varName||`middle_${index}`,
          kind:'middle',
          side:'top',
          meta:item
        }))
      ]
    };
  }
  if(node.type==='signal_source'){
    return {
      inputs:[],
      outputs:[{key:'signal',label:node.props.outputFormat||'输出'}]
    };
  }
  if(node.type==='flow_block'){
    return {
      inputs:[{key:'input',label:node.props.inputName||'输入'}],
      outputs:[{key:'output',label:node.props.outputName||'输出'}]
    };
  }
  if(node.type==='simulation_block'){
    const groups=getSimulationPortSets(node.props);
    return {
      inputs:groups.inputs.map((item,index)=>({
        key:`input-${index}`,
        label:`输入 ${index+1} · ${item.name} / ${item.type}`,
        kind:'input',
        side:'left'
      })),
      outputs:getSimulationFlatOutputs(node.props).map((item,index)=>({
        key:`${item.kind}-${item.index}`,
        label:`${item.kind==='middle'?'中间量':'输出'} ${item.index+1} · ${item.name} / ${item.type}`,
        kind:item.kind,
        side:item.side,
        portIndex:index
      }))
    };
  }
  if(node.type.startsWith('fault_')){
    return {
      inputs:[{key:'input',label:'故障输入'}],
      outputs:[{key:'output',label:'故障输出'}]
    };
  }
  if(node.type.startsWith('instrument_')){
    return {
      inputs:[{key:'input',label:'观测输入'}],
      outputs:[]
    };
  }
  return {inputs:[],outputs:[]};
}

function getPortInfo(node,direction,index=0){
  const ports=getNodePorts(node)[direction==='input'?'inputs':'outputs'];
  const port=ports[index];
  if(!port){return null;}
  const side=port.side||(direction==='input'?'left':'right');
  let localX=direction==='input'?0:node.w;
  let localY=Math.round(node.h/2);
  if(side==='top'){
    const topPorts=ports.filter(item=>(item.side||'right')==='top');
    const topIndex=topPorts.findIndex(item=>item.key===port.key);
    const topGap=node.w/(Math.max(topPorts.length,1)+1);
    localX=Math.round(topGap*(topIndex+1));
    localY=0;
  }else{
    const sidePorts=ports.filter(item=>(item.side||(direction==='input'?'left':'right'))===side);
    const sideIndex=sidePorts.findIndex(item=>item.key===port.key);
    const gap=node.h/(Math.max(sidePorts.length,1)+1);
    localY=Math.round(gap*(sideIndex+1));
    localX=side==='left'?0:node.w;
  }
  return {
    ...port,
    direction,
    side,
    index,
    localX,
    localY,
    x:node.x+localX,
    y:node.y+localY
  };
}

function buildEdgePath(start,end){
  const deltaX=Math.max(64,Math.abs(end.x-start.x)*0.42);
  const deltaY=Math.max(48,Math.abs(end.y-start.y)*0.38);
  const cp1={
    x:start.side==='right'?start.x+deltaX:(start.side==='left'?start.x-deltaX:start.x),
    y:start.side==='top'?start.y-deltaY:start.y
  };
  const cp2={
    x:end.side==='right'?end.x+deltaX:(end.side==='left'?end.x-deltaX:end.x),
    y:end.side==='top'?end.y-deltaY:end.y
  };
  return `M ${start.x} ${start.y} C ${cp1.x} ${cp1.y}, ${cp2.x} ${cp2.y}, ${end.x} ${end.y}`;
}

function clientToStagePoint(clientX,clientY){
  const viewport=getCanvasViewport();
  const rect=viewport.getBoundingClientRect();
  return {
    x:(clientX-rect.left-S.canvasOffsetX)/S.canvasScale,
    y:(clientY-rect.top-S.canvasOffsetY)/S.canvasScale
  };
}

function clampCanvasOffset(){
  const viewport=getCanvasViewport();
  if(!viewport){return;}
  const scaledWidth=CANVAS_STAGE.width*S.canvasScale;
  const scaledHeight=CANVAS_STAGE.height*S.canvasScale;
  const width=viewport.clientWidth;
  const height=viewport.clientHeight;
  const edgeGap=72;

  if(scaledWidth<=width-edgeGap*2){
    S.canvasOffsetX=Math.round((width-scaledWidth)/2);
  }else{
    S.canvasOffsetX=clamp(S.canvasOffsetX,width-scaledWidth-edgeGap,edgeGap);
  }

  if(scaledHeight<=height-edgeGap*2){
    S.canvasOffsetY=Math.round((height-scaledHeight)/2);
  }else{
    S.canvasOffsetY=clamp(S.canvasOffsetY,height-scaledHeight-edgeGap,edgeGap);
  }
}

function applyCanvasTransform(){
  const stage=getCanvasStage();
  if(stage){
    stage.style.transform=`translate(${S.canvasOffsetX}px,${S.canvasOffsetY}px) scale(${S.canvasScale})`;
  }
  const fill=document.getElementById('canvas-zoom-fill');
  const reset=document.getElementById('canvas-zoom-reset');
  const percent=Math.round(S.canvasScale*100);
  if(fill){
    const ratio=clamp((S.canvasScale-CANVAS_STAGE.minScale)/(CANVAS_STAGE.maxScale-CANVAS_STAGE.minScale),0,1);
    fill.style.width=`${Math.round(28+ratio*72)}%`;
  }
  if(reset){
    reset.textContent=`${percent}%`;
  }
}

function setCanvasZoom(nextScale,pivotClientX,pivotClientY){
  const viewport=getCanvasViewport();
  if(!viewport){return;}
  const rect=viewport.getBoundingClientRect();
  const oldScale=S.canvasScale;
  const clampedScale=clamp(nextScale,CANVAS_STAGE.minScale,CANVAS_STAGE.maxScale);
  const pivotX=(pivotClientX ?? (rect.left+rect.width/2))-rect.left;
  const pivotY=(pivotClientY ?? (rect.top+rect.height/2))-rect.top;
  const stageX=(pivotX-S.canvasOffsetX)/oldScale;
  const stageY=(pivotY-S.canvasOffsetY)/oldScale;
  S.canvasScale=clampedScale;
  S.canvasOffsetX=pivotX-stageX*S.canvasScale;
  S.canvasOffsetY=pivotY-stageY*S.canvasScale;
  clampCanvasOffset();
  applyCanvasTransform();
}

  setCanvasZoom(S.canvasScale+0.1);
}

function handleCanvasZoomOut(){
  setCanvasZoom(S.canvasScale-0.1);
}

function handleCanvasZoomReset(){
  S.canvasScale=1;
  S.canvasOffsetX=0;
  S.canvasOffsetY=0;
  clampCanvasOffset();
  applyCanvasTransform();
}

  switch(node.type){
    case 'signal_source':
      return `${props.waveType} · 幅值${props.amplitude} · ${props.outputFormat}`;
    case 'flow_block':
      return `${props.inputName} / ${props.inputFormat} 鈫?${props.outputName} / ${props.outputFormat}${faultNote}`;
    case 'simulation_block':
      normalizeSimulationInterfaces(props);
      return `${props.inputs.length}鍏?路 ${props.outputs.length}鍑?路 ${props.middleVars.length}涓棿 · ${props.moduleType}${faultNote}`;
    case 'fault_bias':
    case 'fault_noise':
    case 'fault_stuck':
    case 'fault_drift':
      return `${props.faultType} · ${props.layer}`;
    case 'instrument_scope':
      return `${props.instrumentType} · 鍙屽嚮查看波形`;
    case 'instrument_spectrum':
    case 'instrument_logger':
      return `${props.instrumentType} · ${props.sampleRate}`;
    case 'link_normal':
    case 'link_can':
      return `${props.sourcePort} 鈫?${props.targetPort}`;
    default:
      return '';
  }
}

function renderEdges(){
  const edgeGroup=document.getElementById('edge-group');
  const preview=document.getElementById('edge-preview');
  if(!edgeGroup||!preview){return;}
  edgeGroup.innerHTML='';

  S.modelEdges.forEach(edge=>{
    const sourceNode=getNode(edge.sourceNodeId);
    const targetNode=getNode(edge.targetNodeId);
    if(!sourceNode||!targetNode){return;}
    const sourcePort=getPortInfo(sourceNode,'output',edge.sourcePortIndex||0);
    const targetPort=getPortInfo(targetNode,'input',edge.targetPortIndex||0);
    if(!sourcePort||!targetPort){return;}
    const selected=S.selEdge===edge.id;
    const faulted=Boolean(edge.injectedFault);
    const edgePath=buildEdgePath(sourcePort,targetPort);

    const hit=document.createElementNS('http://www.w3.org/2000/svg','path');
    hit.setAttribute('d',edgePath);
    hit.setAttribute('class','edge-hit');
    hit.dataset.edgeId=edge.id;
    hit.addEventListener('click',event=>{
      event.stopPropagation();
      selectEdge(edge.id);
    });
    edgeGroup.appendChild(hit);

    const path=document.createElementNS('http://www.w3.org/2000/svg','path');
    path.setAttribute('d',edgePath);
    path.setAttribute('class',`edge-path edge-path--${edge.lineType}${selected?' is-selected':''}${faulted?' is-faulted':''}`);
    path.setAttribute('marker-end',faulted?'url(#edge-arrow-fault)':`url(#edge-arrow-${edge.lineType})`);
    path.dataset.edgeId=edge.id;
    path.addEventListener('click',event=>{
      event.stopPropagation();
      selectEdge(edge.id);
    });
    edgeGroup.appendChild(path);

    const label=document.createElementNS('http://www.w3.org/2000/svg','text');
    label.setAttribute('class',`edge-label${selected?' is-selected':''}${faulted?' is-faulted':''}`);
    label.setAttribute('x',String(Math.round((sourcePort.x+targetPort.x)/2)));
    label.setAttribute('y',String(Math.round((sourcePort.y+targetPort.y)/2)-10));
    label.setAttribute('text-anchor','middle');
    label.dataset.edgeId=edge.id;
    label.textContent=getEdgeLabel(edge);
    label.addEventListener('click',event=>{
      event.stopPropagation();
      selectEdge(edge.id);
    });
    edgeGroup.appendChild(label);
  });

  if(S.pendingConnection){
    const sourceNode=getNode(S.pendingConnection.sourceNodeId);
    const sourcePort=sourceNode?getPortInfo(sourceNode,'output',S.pendingConnection.sourcePortIndex||0):null;
    if(sourcePort){
      preview.style.display='block';
      preview.setAttribute('d',buildEdgePath(sourcePort,{
        x:S.pendingConnection.x,
        y:S.pendingConnection.y
      }));
      preview.setAttribute('stroke',S.pendingConnection.lineType==='can'?'#8c7cf6':'#7ab7ff');
      preview.setAttribute('stroke-width',S.pendingConnection.lineType==='can'?'2.8':'2');
    }else{
      preview.style.display='none';
      preview.setAttribute('d','');
    }
  }else{
    preview.style.display='none';
    preview.setAttribute('d','');
  }
}

  element.style.left=`${node.x}px`;

function beginNodeDrag(event,nodeId){
  if(event.button!==0||event.target.closest('.node-port')){return;}
  const node=getNode(nodeId);
  if(!node){return;}
  if(node.type==='instrument_scope'){
    const now=Date.now();
    if(NODE_DOUBLE_CLICK.nodeId===nodeId&&(now-NODE_DOUBLE_CLICK.timestamp)<320){
      NODE_DOUBLE_CLICK.nodeId='';
      NODE_DOUBLE_CLICK.timestamp=0;
      openScope(nodeId);
      event.preventDefault();
      return;
    }
    NODE_DOUBLE_CLICK.nodeId=nodeId;
    NODE_DOUBLE_CLICK.timestamp=now;
  }
  selectNode(nodeId);
  NODE_DRAG.active=true;
  NODE_DRAG.pointerId=event.pointerId;
  NODE_DRAG.nodeId=nodeId;
  NODE_DRAG.startX=event.clientX;
  NODE_DRAG.startY=event.clientY;
  NODE_DRAG.originX=node.x;
  NODE_DRAG.originY=node.y;

  if(el){
    el.classList.add('is-dragging');
    if(el.setPointerCapture){
      el.setPointerCapture(event.pointerId);
    }
  }
  event.preventDefault();
}

function handlePortClick(event,nodeId,direction,index){
  event.stopPropagation();
  const node=getNode(nodeId);
  if(!node){return;}
  const port=getPortInfo(node,direction,index);
  if(!port){return;}

  if(direction==='output'){
    S.pendingConnection={
      sourceNodeId:nodeId,
      sourcePortIndex:index,
      lineType:S.activeLineType,
      x:port.x,
      y:port.y
    };
    renderEdges();
    return;
  }

  if(!S.pendingConnection){
    toast('请先从输出端口开始连线','w');
    return;
  }

  if(S.pendingConnection.sourceNodeId===nodeId){
    toast('请连接到其他模块的输入端口','w');
    clearPendingConnection();
    return;
  }

  const sourceNode=getNode(S.pendingConnection.sourceNodeId);
  const sourcePort=getPortInfo(sourceNode,'output',S.pendingConnection.sourcePortIndex||0);
  const sourceMeta=buildEdgePortMeta(sourcePort);
  const targetMeta=buildEdgePortMeta(port);
  if(!arePortTypesCompatible(sourceMeta,targetMeta)){
    toast(`端口类型不兼容：${sourceMeta.type||'any'} -> ${targetMeta.type||'any'}`,'w');
    clearPendingConnection();
    return;
  }

  const duplicate=S.modelEdges.some(edge=>
    edge.sourceNodeId===S.pendingConnection.sourceNodeId &&
    edge.targetNodeId===nodeId &&
    (edge.sourcePortIndex||0)===(S.pendingConnection.sourcePortIndex||0) &&
    (edge.targetPortIndex||0)===index
  );
  if(duplicate){
    toast('该连线已存在','w');
    clearPendingConnection();
    return;
  }

  const occupiedTarget=S.modelEdges.some(edge=>
    edge.targetNodeId===nodeId &&
    (edge.targetPortIndex||0)===index
  );
  if(occupiedTarget){
    toast('该输入端口已有连线，请先删除原连线，避免信号耦合','w');
    clearPendingConnection();
    return;
  }

  const newEdge={
    id:`edge-${++S.edgeSeq}`,
    lineType:S.pendingConnection.lineType,
    sourceNodeId:S.pendingConnection.sourceNodeId,
    targetNodeId:nodeId,
    sourcePortIndex:S.pendingConnection.sourcePortIndex||0,
    targetPortIndex:index,
    sourcePort:{
      index:S.pendingConnection.sourcePortIndex||0,
      varName:sourceMeta.varName,
      displayName:sourceMeta.displayName,
      type:sourceMeta.type
    },
    targetPort:{
      index,
      varName:targetMeta.varName,
      displayName:targetMeta.displayName,
      type:targetMeta.type
    },
    sourcePortMeta:sourceMeta,
    targetPortMeta:targetMeta
  };
  S.modelEdges.push(newEdge);
  syncAllPythonInputConnectionState();
  markTopologyDirty(getMutationScopeForEdge(newEdge));
  S.selEdge=newEdge.id;
  S.selBlk=null;
  clearPendingConnection();
  renderPropertyPanel(newEdge);
  updateUI();
  toast('连线已创建','s');
}

  S.selEdge=null;
  renderModelNodes();
  renderPropertyPanel(node);
}

  const edge=getEdge(id);
  if(!edge){return;}
  if(S.pendingConnection){
    S.pendingConnection=null;
  }
  S.selEdge=id;
  S.selBlk=null;
  renderModelNodes();
  renderPropertyPanel(edge);
}

function deleteSelectedNode(){
  const node=getNode(S.selBlk);
  if(!node){return;}
  const removedName=node.props.name;
  const removedEdges=S.modelEdges.filter(edge=>edge.sourceNodeId===node.id||edge.targetNodeId===node.id).length;
  const protocolEdgeAffected=S.protocolInjectTargetId&&S.modelEdges.some(edge=>edge.id===S.protocolInjectTargetId&&(edge.sourceNodeId===node.id||edge.targetNodeId===node.id));

  S.modelNodes=S.modelNodes.filter(item=>item.id!==node.id);
  S.modelEdges=S.modelEdges.filter(edge=>edge.sourceNodeId!==node.id&&edge.targetNodeId!==node.id);
  S.faultedBlks=S.faultedBlks.filter(id=>id!==node.id);
  if(S.injectTargetId===node.id){
    S.injectTargetId='';
  }
  if(protocolEdgeAffected){
    S.protocolInjectTargetId='';
  }
  if(S.pendingConnection&&S.pendingConnection.sourceNodeId===node.id){
    S.pendingConnection=null;
  }
  S.selBlk=null;
  markTopologyDirty(getMutationScopeForType(node.type));
  syncAllPythonInputConnectionState();
  renderModelNodes();
  renderPropertyPanel(null);
  syncConfigTargets();
  updateUI();
  toast(`模块“${removedName}”已删除${removedEdges?`，并移除 ${removedEdges} 条连线`:''}`,'s');
}

function deleteSelectedEdge(){
  const edge=getEdge(S.selEdge);
  if(!edge){return;}
  const edgeName=edge.lineType==='can'?'CAN 总线':'普通连接线';
  S.modelEdges=S.modelEdges.filter(item=>item.id!==edge.id);
  if(S.protocolInjectTargetId===edge.id){
    S.protocolInjectTargetId='';
    S.protocolInjectSelectionMode='';
    S.protocolInjectSelectionId='';
  }
  S.selEdge=null;
  syncAllPythonInputConnectionState();
  markTopologyDirty(getMutationScopeForEdge(edge));
  renderModelNodes();
  renderPropertyPanel(null);
  updateUI();
  toast(`连线“${edgeName}”已删除`,'s');
}

function deleteCurrentSelection(){
  if(S.selEdge){
    deleteSelectedEdge();
    return;
  }
  if(S.selBlk){
    deleteSelectedNode();
  }
}

function syncConfigTargets(preferredId=''){
  const sel=document.getElementById('cftgt');
  if(!sel){return;}
  const currentValue=preferredId||sel.value;
  const nodes=getConfigurableNodes();
  if(nodes.length===0){
    sel.innerHTML='<option value="">璇峰厛鍒涘缓系统建模缁勪欢</option>';
    return;
  }
  sel.innerHTML=nodes.map(node=>`<option value="${node.id}">${escapeHtml(node.props.name)} · ${COMPONENT_LIBRARY[node.type].label}</option>`).join('');
  sel.value=nodes.some(node=>node.id===currentValue)?currentValue:nodes[0].id;
}

  if(S.selBlk&&getNode(S.selBlk)){renderPropertyPanel(getNode(S.selBlk));}

function createNode(type,rawX,rawY){
  const meta=COMPONENT_LIBRARY[type];
  if(!meta){return;}
  const props=cloneDefaults(meta.defaults);
  const count=getNodeTypeCount(type);
  if(type==='signal_source'){props.name=`信号源 ${count}`;}
  if(type==='flow_block'){props.name=`流程块 ${count}`;}
  if(type==='simulation_block'){props.name=`仿真块 ${count}`;}
  if(type.startsWith('fault_')){props.name=`${meta.label} ${count}`;}
  if(type.startsWith('instrument_')){props.name=`${meta.label} ${count}`;}
  if(type.startsWith('link_')){props.name=`${meta.label} ${count}`;}

  if(type==='simulation_block'){
    normalizeSimulationInterfaces(props);
  }
  const size=type==='simulation_block'?getSimulationNodeSize(props):{width:meta.width,height:meta.height};
  const point={
    x:clamp(Math.round(rawX-size.width/2),CANVAS_STAGE.nodePadding,CANVAS_STAGE.width-size.width-CANVAS_STAGE.nodePadding),
    y:clamp(Math.round(rawY-size.height/2),CANVAS_STAGE.nodeMinY,CANVAS_STAGE.height-size.height-CANVAS_STAGE.nodePadding)
  };
  S.modelNodes.push({
    id:`node-${++S.nodeSeq}`,
    type,
    x:point.x,
    y:point.y,
    w:size.width,
    h:size.height,
    props,
    pythonBinding:type==='simulation_block'?createDefaultPythonBinding():null
  });
  markTopologyDirty(getMutationScopeForType(type));
  const node=S.modelNodes[S.modelNodes.length-1];
  renderModelNodes();
  syncConfigTargets(node.id);
  selectNode(node.id);
  updateUI();
  toast(`${meta.label} 已加入画布`,'s');
}

  });
}

function bindCanvasInteractions(){
  const viewport=getCanvasViewport();
  if(!viewport){return;}

  viewport.addEventListener('wheel',event=>{
    if(!S.sysLoaded){return;}
    event.preventDefault();
    setCanvasZoom(S.canvasScale+(event.deltaY<0?0.08:-0.08),event.clientX,event.clientY);
  },{passive:false});

  viewport.addEventListener('pointerdown',event=>{
    if(event.button!==0){return;}
    const target=event.target;
    const edgeInteractive=target?.closest?.('.edge-hit, .edge-path, .edge-label');
    const edgeId=edgeInteractive?.dataset?.edgeId||findNearestEdgeIdByClientPoint(event.clientX,event.clientY);
    if(target.closest('.blk')||target.closest('.node-port')||edgeInteractive){return;}
    if(edgeId){
      selectEdge(edgeId);
      return;
    }
    if(S.pendingConnection){
      clearPendingConnection();
      return;
    }
    if(S.selBlk||S.selEdge){
      clearCanvasSelection();
    }
    CANVAS_PAN.active=true;
    CANVAS_PAN.pointerId=event.pointerId;
    CANVAS_PAN.startX=event.clientX;
    CANVAS_PAN.startY=event.clientY;
    CANVAS_PAN.originX=S.canvasOffsetX;
    CANVAS_PAN.originY=S.canvasOffsetY;
    viewport.classList.add('panning');
    if(viewport.setPointerCapture){
      viewport.setPointerCapture(event.pointerId);
    }
  });

  viewport.addEventListener('pointermove',event=>{
    if(S.pendingConnection){
      const point=clientToStagePoint(event.clientX,event.clientY);
      S.pendingConnection.x=point.x;
      S.pendingConnection.y=point.y;
      renderEdges();
    }

    if(NODE_DRAG.active&&NODE_DRAG.pointerId===event.pointerId){
      const node=getNode(NODE_DRAG.nodeId);
      if(!node){return;}
      const deltaX=(event.clientX-NODE_DRAG.startX)/S.canvasScale;
      const deltaY=(event.clientY-NODE_DRAG.startY)/S.canvasScale;
      node.x=clamp(Math.round(NODE_DRAG.originX+deltaX),CANVAS_STAGE.nodePadding,CANVAS_STAGE.width-node.w-CANVAS_STAGE.nodePadding);
      node.y=clamp(Math.round(NODE_DRAG.originY+deltaY),CANVAS_STAGE.nodeMinY,CANVAS_STAGE.height-node.h-CANVAS_STAGE.nodePadding);
      updateNodePositionDOM(node);
      renderEdges();
      return;
    }

    if(CANVAS_PAN.active&&CANVAS_PAN.pointerId===event.pointerId){
      S.canvasOffsetX=CANVAS_PAN.originX+(event.clientX-CANVAS_PAN.startX);
      S.canvasOffsetY=CANVAS_PAN.originY+(event.clientY-CANVAS_PAN.startY);
      clampCanvasOffset();
      applyCanvasTransform();
    }
  });

  const endPointerInteraction=event=>{
    if(NODE_DRAG.active&&NODE_DRAG.pointerId===event.pointerId){
      const draggedNode=getNode(NODE_DRAG.nodeId);
      const draggedEl=document.getElementById(`b-${NODE_DRAG.nodeId}`);
      if(draggedEl){
        draggedEl.classList.remove('is-dragging');
      }
      NODE_DRAG.active=false;
      NODE_DRAG.pointerId=null;
      NODE_DRAG.nodeId='';
      if(draggedNode){renderEdges();}
    }

    if(CANVAS_PAN.active&&CANVAS_PAN.pointerId===event.pointerId){
      CANVAS_PAN.active=false;
      CANVAS_PAN.pointerId=null;
      viewport.classList.remove('panning');
    }
  };

  viewport.addEventListener('pointerup',endPointerInteraction);
  viewport.addEventListener('pointercancel',endPointerInteraction);

  const runtimeListeners=getRuntimeListenerStore();
  if(runtimeListeners.canvasWindowKeydown){
    window.removeEventListener('keydown',runtimeListeners.canvasWindowKeydown);
  }
  runtimeListeners.canvasWindowKeydown=event=>{
    if(event.key==='Escape'&&S.pendingConnection){
      clearPendingConnection();
      return;
    }
    const active=document.activeElement;
    const editing=active&&(
      active.tagName==='INPUT'||
      active.tagName==='TEXTAREA'||
      active.tagName==='SELECT'||
      active.isContentEditable
    );
    if((event.key==='Delete'||event.key==='Backspace')&&!editing&&!document.querySelector('.overlay.open')&&(S.selBlk||S.selEdge)){
      event.preventDefault();
      deleteCurrentSelection();
    }
  };
  window.addEventListener('keydown',runtimeListeners.canvasWindowKeydown);

  if(runtimeListeners.canvasWindowResize){
    window.removeEventListener('resize',runtimeListeners.canvasWindowResize);
  }
  runtimeListeners.canvasWindowResize=()=>{
    clampCanvasOffset();
    applyCanvasTransform();
  };
  window.addEventListener('resize',runtimeListeners.canvasWindowResize);
}

function buildPythonBindingPortRows(items,kind){
  if(!items.length){
    return '<div class="props-help">当前娌℃湁鍙敤端口</div>';
  }
  return items.map(item=>{
    const status=kind==='inputs'
      ?(item.connected?'<span class="pv pv-ok">已连接</span>':'<span class="pv pv-fault">鏈繛鎺?</span>')
      :'<span class="pv pv-ok">已暴露</span>';
    const title=escapeHtml(item.displayName||item.varName||'-');
    const type=escapeHtml(item.type||'any');
    const note=item.comment?`<div class="props-help">${escapeHtml(item.comment)}</div>`:'';
    const defaultRow=kind==='inputs'&&item.default!==null&&item.default!==undefined&&item.default!==''
      ?`<div class="prow"><span class="pk">榛樿鍊?</span><span class="pv">${escapeHtml(item.default)}<</span></div>`
      :'';
    return `
      <div class="pgroup pgroup--compact">
        <div class="prow"><span class="pk">${title}<</span>${status}</div>
        <div class="prow"><span class="pk">类型<</span><span class="pv">${type}<</span></div>
        ${defaultRow}
        ${note}
      </div>`;
  }).join('');
}

function doImportSys(){
  const btn=document.getElementById('btn-imp-sys');
  btn.textContent='导入中…';
  btn.disabled=true;
  setTimeout(()=>{
    let restored=false;
    btn.textContent='导入系统模型';
    btn.disabled=false;
    S.sysLoaded=true;
    S.step=1;
    document.getElementById('empty').style.display='none';
    document.getElementById('diagram').classList.add('on');
    handleCanvasZoomReset();
    try{
      const raw=window.localStorage?.getItem(WORKBENCH_STORAGE_KEY);
      if(raw){
        restoreSystemModelSnapshot(JSON.parse(raw));
        S.systemSaved=true;
        S.step=S.modelNodes.length?2:1;
        restored=S.modelNodes.length>0;
      }else{
        clearPendingConnection();
        renderModelNodes();
        renderPropertyPanel(null);
      }
    }catch(error){
      console.warn('Failed to restore saved system model',error);
      clearPendingConnection();
      renderModelNodes();
      renderPropertyPanel(null);
    }
    toast(restored?'已恢复已保存的系统模型':'系统建模画布已就绪，请拖拽组件开始建模','s');
    updateUI();
  },1000);
}

function doSaveSys(){
  if(!S.sysLoaded){toast('请先导入系统模型','w');return;}
  if(S.modelNodes.length===0){toast('请至少添加一个组件后再保存','w');return;}
  try{
    const snapshot=createSystemModelSnapshot();
    window.localStorage?.setItem(WORKBENCH_STORAGE_KEY,JSON.stringify(snapshot));
  }catch(error){
    console.warn('Failed to persist system model snapshot',error);
    toast('系统模型保存失败，请稍后重试','w');
    return;
  }
  S.systemSaved=true;
  S.step=2;
  toast('系统模型已保存','s');
  updateUI();
}

function getIncomingEdgesForNode(nodeId){
  return S.modelEdges
    .filter(edge=>edge.targetNodeId===nodeId)

}

function getOutgoingEdgesForNode(nodeId){
  return S.modelEdges.filter(edge=>edge.sourceNodeId===nodeId);
}

function noiseAt(seed,stepIndex,time){
  const raw=Math.sin(seed*12.9898+stepIndex*78.233+time*5.173)*43758.5453;
  const frac=raw-Math.floor(raw);
  return frac*2-1;
}

function parseScalar(value,fallback=0){
  const parsed=Number.parseFloat(String(value).replace(/[^\d+\-.eE]/g,''));
  return Number.isFinite(parsed)?parsed:fallback;
}

function ensureScopeSelection(preferredId=''){
  if(preferredId&&getNode(preferredId)?.type==='instrument_scope'){
    SIM.activeScopeId=preferredId;
    return;
  }
  if(getNode(SIM.activeScopeId)?.type==='instrument_scope'){return;}

}

function ensureNodeRuntime(bucket,node){
  if(!bucket.nodeStates[node.id]){
    bucket.nodeStates[node.id]={
      lastOutput:0,
      lastInput:0,
      lastOutputs:[0],
      lastInputValues:[],
      lastMiddleValues:[],
      prevInput:0,
      prevAggregateInput:0,
      integrator:0,
      firstOrder:0
    };
  }
  return bucket.nodeStates[node.id];
}

function ensureEdgeRuntime(bucket,edge){
  if(!bucket.edgeStates[edge.id]){
    bucket.edgeStates[edge.id]={lastOutput:0,delayQueue:[],history:[]};
  }
  return bucket.edgeStates[edge.id];
}

function findInputCouplingConflicts(){
  const seen=new Map();
  const conflicts=[];
  S.modelEdges.forEach(edge=>{
    const key=`${edge.targetNodeId}:${edge.targetPortIndex||0}`;
    if(seen.has(key)){
      conflicts.push(seen.get(key),edge);
    }else{
      seen.set(key,edge);
    }
  });
  return [...new Map(conflicts.map(edge=>[edge.id,edge])).values()];
}

function computeSourceSignal(node,time){
  const amplitude=parseScalar(node.props.amplitude,1);
  const frequency=parseScalar(node.props.frequency,1);
  if(node.props.waveType==='闃惰穬'){
    const stepTime=Math.max(0.2,frequency);
    return time>=stepTime?amplitude:0;
  }
  if(node.props.waveType==='常值'){
    return amplitude;
  }
  return amplitude*Math.sin(2*Math.PI*frequency*time);
}

function applyElectricalFault(node,inputValue,outputValue,time){
  if(!node.injectedFault){return outputValue;}
  const name=node.injectedFault.name||'';
  if(name.includes('过流')){
    return Math.abs(inputValue)>1.2?0:outputValue*0.78;
  }
  if(name.includes('短路')||name.includes('断路')||name.includes('开路')){
    return 0;
  }
  if(name.includes('鐢靛帇')){
    return outputValue*(0.94+0.08*Math.sin(time*7.4));
  }
  return outputValue*0.85;
}

function applyPhysicalFaultNode(node,inputValue,time,modeKey){
  if(modeKey==='reference'){return inputValue;}
  if(node.type==='fault_bias'){
    return inputValue+0.5;
  }
  if(node.type==='fault_noise'){
    return inputValue+noiseAt(getNodeSeed(node.id),SIM.stepIndex,time)*0.22;
  }
  return inputValue;
}

function flipBitValue(value){
  const scaled=Math.round(value*256);
  return (scaled^0b1000)/256;
}

function transmitEdgeSignal(edge,value,modeKey,time){
  const bucket=modeKey==='reference'?SIM.reference:SIM.actual;
  const edgeState=ensureEdgeRuntime(bucket,edge);
  const base=Number.isFinite(value)?value:0;
  let nextValue=base;

  if(modeKey==='actual'&&edge.injectedFault?.layer==='protocol'){
    const kind=edge.injectedFault.faultKind||'';
    const faultCode=edge.injectedFault.faultCode||'';
    const held=Number.isFinite(edgeState.lastOutput)?edgeState.lastOutput:0;
    if(faultCode==='loss'||kind==='丢包'){
      nextValue=noiseAt(getNodeSeed(edge.id),SIM.stepIndex,time)>0.55?held:base;
    }else if(faultCode==='delay'||kind==='延迟'){
      const delaySteps=clamp(Math.round(0.35/Math.max(SIM.stepSize,0.001)),1,8);
      edgeState.delayQueue.push(base);
      nextValue=edgeState.delayQueue.length>delaySteps?edgeState.delayQueue.shift():held;
    }else if(faultCode==='bitflip'||kind==='位翻转'){
      nextValue=flipBitValue(base);
    }else if(faultCode==='replay'||kind==='重放'){
      if(edgeState.history.length>6&&SIM.stepIndex>6&&SIM.stepIndex%10===0){
        nextValue=edgeState.history[Math.max(0,edgeState.history.length-6)];
      }else{
        nextValue=base;
      }
    }
  }

  edgeState.lastOutput=nextValue;
  pushLimitedSample(edgeState.history,base,40);
  return nextValue;
}

function sanitizeSignalValue(value){
  return Number.isFinite(value)?value:0;
}

function resolveNodeInputValues(node,modeKey,time,dt,cache,visiting){
  const inputPorts=getNodePorts(node).inputs;
  if(!inputPorts.length){return [];}
  const incoming=getIncomingEdgesForNode(node.id);
  return inputPorts.map((_,portIndex)=>{
    const edge=incoming.find(item=>(item.targetPortIndex||0)===portIndex);
    if(!edge){return 0;}
    const upstream=resolveNodeOutput(edge.sourceNodeId,edge.sourcePortIndex||0,modeKey,time,dt,cache,visiting);
    return transmitEdgeSignal(edge,upstream,modeKey,time);
  });
}

function buildSimulationResult(node,nodeState,inputValues,modeKey,time,dt){
  normalizeSimulationInterfaces(node.props);
  const primaryInput=sanitizeSignalValue(inputValues[0]||0);
  const aggregateInput=inputValues.length
    ?inputValues.reduce((sum,value)=>sum+sanitizeSignalValue(value),0)
    :0;
  const activeInput=inputValues.length>1?aggregateInput:primaryInput;
  let coreOutput=0;
  let stateValue=0;

  if(node.props.moduleType==='寰垎'){
    coreOutput=(activeInput-(nodeState.prevAggregateInput||0))/Math.max(dt,0.001);
    stateValue=activeInput;
  }else if(node.props.moduleType==='绉垎'){
    nodeState.integrator=(nodeState.integrator||0)+activeInput*dt;
    coreOutput=nodeState.integrator;
    stateValue=nodeState.integrator;
  }else{
    const tau=1;
    const previous=nodeState.firstOrder||0;
    const alpha=dt/(tau+dt);
    coreOutput=previous+alpha*(activeInput-previous);
    nodeState.firstOrder=coreOutput;
    stateValue=nodeState.firstOrder;
  }

  nodeState.prevInput=primaryInput;
  nodeState.prevAggregateInput=activeInput;
  if(modeKey==='actual'){
    coreOutput=applyElectricalFault(node,activeInput,coreOutput,time);
  }

  const outputValues=node.props.outputs.map((_,index)=>{
    if(index===0){return sanitizeSignalValue(coreOutput);}
    if(index===1){return sanitizeSignalValue(stateValue);}
    if(index===2){return sanitizeSignalValue(aggregateInput);}
    return sanitizeSignalValue(coreOutput+(index*0.12*aggregateInput));
  });

  const middleValues=node.props.middleVars.map((_,index)=>{
    if(index===0){return sanitizeSignalValue(primaryInput);}
    if(index===1){return sanitizeSignalValue(aggregateInput);}
    if(index===2){return sanitizeSignalValue(stateValue);}
    if(index===3){return sanitizeSignalValue(coreOutput);}
    return sanitizeSignalValue((coreOutput+aggregateInput+primaryInput)/(index+1));
  });

  return {outputs:outputValues,middleValues};
}

function getPythonBindingDefaultValue(port){
  return window.__GZ_PYTHON_RUNTIME__?.getPythonBindingDefaultValue
    ?window.__GZ_PYTHON_RUNTIME__.getPythonBindingDefaultValue(port)
    :0;
}

function readPythonExecutionValue(collection,item,index){
  if(Array.isArray(collection)){
    return collection[index];
  }
  if(!collection||typeof collection!=='object'){
    return undefined;
  }
  if(item?.varName&&Object.prototype.hasOwnProperty.call(collection,item.varName)){
    return collection[item.varName];
  }
  if(item?.displayName&&Object.prototype.hasOwnProperty.call(collection,item.displayName)){
    return collection[item.displayName];
  }
  return undefined;
}

function executeBoundFlowBlock(node,nodeState,inputValues,modeKey,time,dt){
  const binding=ensureFlowNodePythonBinding(node);
  const runtime=window.__GZ_PYTHON_RUNTIME__;
  if(!binding.bound||typeof runtime?.executeFlowBlockPythonBindingSync!=='function'){
    const passthrough=sanitizeSignalValue(inputValues[0]||0);
    return {outputs:[passthrough],middleValues:[]};
  }

  const execution=runtime.executeFlowBlockPythonBindingSync({
    nodeId:node.id,
    binding,
    inputValues,
    mode:modeKey,
    time,
    dt,
    adapterMode:binding.executionMode||'mock',
    adapterEndpoint:binding.executionConfig?.endpoint,
    applyElectricalFault:value=>applyElectricalFault(node,inputValues[0]||0,value,time)
  });

  nodeState.lastPythonPayload=execution.payload||null;
  nodeState.lastPythonResult=execution.response||null;
  return {
    outputs:(execution.outputs||[]).map(sanitizeSignalValue),
    middleValues:(execution.middleValues||[]).map(sanitizeSignalValue)
  };
}

function resolveNodeOutputs(nodeId,modeKey,time,dt,cache,visiting){
  if(cache.has(nodeId)){return cache.get(nodeId);}
  const node=getNode(nodeId);
  if(!node){
    return {outputs:[0],middleValues:[]};
  }
  const bucket=modeKey==='reference'?SIM.reference:SIM.actual;
  const nodeState=ensureNodeRuntime(bucket,node);
  if(visiting.has(nodeId)){
    return {
      outputs:nodeState.lastOutputs?.length?nodeState.lastOutputs:[nodeState.lastOutput||0],
      middleValues:nodeState.lastMiddleValues||[]
    };
  }
  visiting.add(nodeId);
  const inputValues=resolveNodeInputValues(node,modeKey,time,dt,cache,visiting);
  let resolved={outputs:[0],middleValues:[]};

  if(node.type==='signal_source'){
    resolved={outputs:[computeSourceSignal(node,time)],middleValues:[]};
  }else if(node.type==='flow_block'){
    resolved=isPythonBoundFlowBlock(node)
      ?executeBoundFlowBlock(node,nodeState,inputValues,modeKey,time,dt)
      :{outputs:[sanitizeSignalValue(inputValues[0]||0)],middleValues:[]};
  }else if(node.type==='simulation_block'){
    resolved=buildSimulationResult(node,nodeState,inputValues,modeKey,time,dt);
  }else if(node.type.startsWith('fault_')){
    resolved={outputs:[applyPhysicalFaultNode(node,sanitizeSignalValue(inputValues[0]||0),time,modeKey)],middleValues:[]};
  }else if(node.type.startsWith('instrument_')){
    resolved={outputs:[sanitizeSignalValue(inputValues[0]||0)],middleValues:[]};
  }else{
    resolved={outputs:[sanitizeSignalValue(inputValues[0]||0)],middleValues:[]};
  }

  resolved.outputs=(resolved.outputs||[0]).map(sanitizeSignalValue);
  resolved.middleValues=(resolved.middleValues||[]).map(sanitizeSignalValue);
  nodeState.lastInputValues=inputValues.slice();
  nodeState.lastInput=inputValues[0]||0;
  nodeState.lastOutputs=resolved.outputs.slice();
  nodeState.lastMiddleValues=resolved.middleValues.slice();
  nodeState.lastOutput=nodeState.lastOutputs[0]||0;
  visiting.delete(nodeId);
  cache.set(nodeId,resolved);
  return resolved;
}

function resolveNodeOutput(nodeId,outputIndex,modeKey,time,dt,cache,visiting){
  const node=getNode(nodeId);
  const resolved=resolveNodeOutputs(nodeId,modeKey,time,dt,cache,visiting);
  if(!node){return 0;}
  if(node.type==='simulation_block'||(node.type==='flow_block'&&isPythonBoundFlowBlock(node))){
    const mainCount=node.type==='simulation_block'
      ?(normalizeSimulationInterfaces(node.props),node.props.outputs.length)
      :(node.pythonBinding?.portMapping?.outputs?.length||0);
    if(outputIndex<mainCount){
      return sanitizeSignalValue(resolved.outputs[outputIndex]??resolved.outputs[0]??0);
    }
    return sanitizeSignalValue(resolved.middleValues[outputIndex-mainCount]??resolved.outputs[0]??0);
  }
  return sanitizeSignalValue(resolved.outputs[outputIndex]??resolved.outputs[0]??0);
}

function resolveNodeInput(node,modeKey,time,dt,cache,visiting){
  return resolveNodeInputValues(node,modeKey,time,dt,cache,visiting)[0]||0;
}

function resolveNodeSignal(nodeId,modeKey,time,dt,cache,visiting){
  return resolveNodeOutput(nodeId,0,modeKey,time,dt,cache,visiting);
}

function recordScopeSamples(time,dt){
  const actualCache=new Map();
  const referenceCache=new Map();
  const actualVisiting=new Set();
  const referenceVisiting=new Set();
  getScopeNodes().forEach(scopeNode=>{
    const actualValue=resolveNodeSignal(scopeNode.id,'actual',time,dt,actualCache,actualVisiting);
    const referenceValue=resolveNodeSignal(scopeNode.id,'reference',time,dt,referenceCache,referenceVisiting);
    pushLimitedSample(SIM.actual.scopeSamples[scopeNode.id],{t:time,v:actualValue});
    pushLimitedSample(SIM.reference.scopeSamples[scopeNode.id],{t:time,v:referenceValue});
  });
}

function runSimulationTick(){
  if(!SIM.inited){return false;}
  if(SIM.time>=SIM.duration){return true;}
  const time=SIM.time;
  const dt=SIM.stepSize;
  recordScopeSamples(time,dt);
  SIM.stepIndex+=1;
  SIM.time=Number((SIM.time+dt).toFixed(6));
  SIM.hasSamples=true;
  S.step=3;
  if(document.getElementById('ov-scope').classList.contains('open')){
    drawScope();
  }
  updateUI();
  return SIM.time>=SIM.duration;
}

function startSimulationLoop(){
  clearSimLoop();
  SIM.timer=setInterval(()=>{
    if(!SIM.running||SIM.paused){return;}
    const completed=runSimulationTick();
    if(completed){
      simStop('completed');
    }
  },getSimulationLoopInterval());
}

function buildSpectrum(values,binCount=24){
  if(values.length<4){return new Array(binCount).fill(0);}
  const bins=[];
  for(let k=1;k<=binCount;k++){
    let real=0;
    let imag=0;
    values.forEach((value,index)=>{
      const angle=2*Math.PI*k*index/values.length;
      real+=value*Math.cos(angle);
      imag-=value*Math.sin(angle);
    });
    bins.push(Math.sqrt(real*real+imag*imag)/values.length);
  }
  const maxValue=Math.max(...bins,1e-6);
  return bins.map(value=>value/maxValue);
}

  const ctx2=ch2.getContext('2d');
  if(!scopeNode){
    drawScopeEmpty(ctx1,ch1.width,ch1.height,'请先在画布中放置示波器');
    drawScopeEmpty(ctx2,ch2.width,ch2.height,'绛夊緟频谱数据');
    return;
  }

  const actualSeries=(SIM.actual.scopeSamples[scopeNode.id]||[]).slice(-240);
  const referenceSeries=(SIM.reference.scopeSamples[scopeNode.id]||[]).slice(-240);
  if(actualSeries.length===0){
    drawScopeEmpty(ctx1,ch1.width,ch1.height,'双击示波器后，开始仿真即可看到波形');
    drawScopeEmpty(ctx2,ch2.width,ch2.height,'鏆傛棤频谱数据');
    return;
  }

  drawScopeGrid(ctx1,ch1.width,ch1.height);
  const visibleSeries=S.cmpMode&&referenceSeries.length?referenceSeries.concat(actualSeries):actualSeries;
  let minValue=Math.min(...visibleSeries.map(item=>item.v));
  let maxValue=Math.max(...visibleSeries.map(item=>item.v));
  if(Math.abs(maxValue-minValue)<1e-6){
    maxValue+=1;
    minValue-=1;
  }else{
    const padding=(maxValue-minValue)*0.18;
    maxValue+=padding;
    minValue-=padding;
  }
  const startTime=actualSeries[0].t;
  const endTime=actualSeries[actualSeries.length-1].t||startTime+SIM.stepSize;
  const span=Math.max(endTime-startTime,SIM.stepSize);
  const getX=time=>18+((time-startTime)/span)*(ch1.width-36);
  const getY=value=>12+(maxValue-value)/(maxValue-minValue)*(ch1.height-24);
  const zeroY=getY(0);
  if(zeroY>=8&&zeroY<=ch1.height-8){
    ctx1.strokeStyle='#c2d6ea';
    ctx1.lineWidth=0.8;
    ctx1.beginPath();
    ctx1.moveTo(0,zeroY);
    ctx1.lineTo(ch1.width,zeroY);
    ctx1.stroke();
  }
  if(S.cmpMode&&referenceSeries.length){
    drawWaveSeries(ctx1,ch1.width,ch1.height,referenceSeries,'#1d6fbf',getX,getY);
    drawWaveSeries(ctx1,ch1.width,ch1.height,actualSeries,'#dc2626',getX,getY);
  }else{
    drawWaveSeries(ctx1,ch1.width,ch1.height,actualSeries,'#1d6fbf',getX,getY);
  }
  ctx1.fillStyle='#94b4cc';
  ctx1.font='8px JetBrains Mono,monospace';
  ctx1.fillText(`${startTime.toFixed(1)}s`,4,ch1.height-4);
  ctx1.fillText(`${endTime.toFixed(1)}s`,ch1.width-32,ch1.height-4);

  drawScopeGrid(ctx2,ch2.width,ch2.height);
  const actualSpectrum=buildSpectrum(actualSeries.map(item=>item.v));
  const referenceSpectrum=buildSpectrum(referenceSeries.map(item=>item.v));
  const binWidth=ch2.width/actualSpectrum.length;
  actualSpectrum.forEach((value,index)=>{
    const x=index*binWidth;
    const height=value*(ch2.height-18);
    const gradient=ctx2.createLinearGradient(x,ch2.height-height,x,ch2.height);
    gradient.addColorStop(0,S.cmpMode?'rgba(220,38,38,0.85)':'rgba(29,111,191,0.88)');
    gradient.addColorStop(1,'rgba(74,158,255,0.08)');
    ctx2.fillStyle=gradient;
    ctx2.fillRect(x+1,ch2.height-height,Math.max(binWidth-2,2),height);
    if(S.cmpMode&&referenceSpectrum[index]!=null){
      const refHeight=referenceSpectrum[index]*(ch2.height-18);
      ctx2.fillStyle='rgba(29,111,191,0.24)';
      ctx2.fillRect(x+1,ch2.height-refHeight,Math.max(binWidth-2,2),refHeight);
    }
  });
  ctx2.fillStyle='#94b4cc';
  ctx2.font='8px JetBrains Mono,monospace';
  ctx2.fillText('0Hz',3,ch2.height-4);
  ctx2.fillText(`${(0.5/Math.max(SIM.stepSize,0.001)).toFixed(1)}Hz`,Math.max(ch2.width-42,3),ch2.height-4);
}

function simInit(silent=false){
  if(!validateSimulationTopology()){return;}
  const autoSaved=!S.systemSaved;
  if(autoSaved){
    S.systemSaved=true;
    if(S.step<2){S.step=2;}
  }
  SIM.datasetName=document.getElementById('sim-name').value.trim()||'test';
  SIM.duration=Math.max(parseScalar(document.getElementById('sim-dur').value,10),0.1);
  SIM.stepSize=Math.max(parseScalar(document.getElementById('sim-step').value,0.1),0.001);
  SIM.realtime=document.getElementById('sim-rt').checked;
  SIM.inited=true;
  SIM.running=false;
  SIM.paused=false;
  resetSimulationRuntime();
  updateSimbarEnabled();
  updateUI();
  if(document.getElementById('ov-scope').classList.contains('open')){
    drawScope();
  }
  if(!silent){
    if(autoSaved){
      toast(`系统妯″瀷宸茶嚜鍔ㄤ繚瀛橈紝骞跺畬鎴愪豢鐪熷垵濮嬪寲锛?{SIM.datasetName} · 鏃堕暱 ${SIM.duration}s · 姝ラ暱 ${SIM.stepSize}s`,'s');
      return;
    }
    toast(`仿真鍒濆鍖栧畬鎴愶細${SIM.datasetName} · 鏃堕暱 ${SIM.duration}s · 姝ラ暱 ${SIM.stepSize}s`,'s');
  }
}

function simRun(){
  if(!SIM.inited){
    simInit();
    if(!SIM.inited){return;}
  }
  if(SIM.time>=SIM.duration){
    simInit(true);
  }
  SIM.realtime=document.getElementById('sim-rt').checked;
  SIM.running=true;
  SIM.paused=false;
  startSimulationLoop();
  updateSimbarEnabled();
  updateUI();
  toast('仿真开始运行','s');
}

function simStep(){
  if(!SIM.inited){
    simInit();
    if(!SIM.inited){return;}
  }
  if(SIM.time>=SIM.duration){
    toast('当前仿真宸茬粨鏉燂紝璇烽噸鏂板垵濮嬪寲鍚庡啀姝ヨ繘','w');
    return;
  }
  const completed=runSimulationTick();
  updateSimbarEnabled();
  if(completed){
    simStop('completed');
  }else{
    toast(`姝ヨ繘瀹屾垚 · t = ${SIM.time.toFixed(2)}s`,'s');
  }
}

function simPause(){
  if(!SIM.running){return;}
  SIM.paused=!SIM.paused;
  if(SIM.paused){
    clearSimLoop();
    toast('仿真已暂停','w');
  }else{
    startSimulationLoop();
    toast('仿真继续杩愯','s');
  }
  updateSimbarEnabled();
  updateUI();
}

function simStop(mode='manual'){
  clearSimLoop();
  const hasData=SIM.hasSamples;
  SIM.running=false;
  SIM.paused=false;
  S.simDone=hasData;
  if(hasData){
    S.step=3;
  }
  updateSimbarEnabled();
  updateUI();
  if(mode==='completed'){
    toast('仿真已完成，结果可保存','s');
  }else{
    toast(hasData?'仿真已终止，结果可保存':'仿真已终止','s');
  }
}

function onRtToggle(el){
  SIM.realtime=el.checked;
  if(SIM.running&&!SIM.paused){
    startSimulationLoop();
  }
  toast(el.checked?'实时模式已开启':'实时模式已关闭', el.checked?'s':'w');
}

function buildPythonBindingPortRows(items,kind){
  if(!items.length){
    return '<div class="props-help">当前娌℃湁鍙敤端口銆?</div>';
  }
  return items.map(item=>{
    const status=kind==='inputs'
      ?(item.connected?'<span class="pv pv-ok">已连接</span>':'<span class="pv pv-fault">鏈繛鎺?</span>')
      :'<span class="pv pv-ok">已暴露</span>';
    const title=escapeHtml(item.displayName||item.varName||'-');
    const type=escapeHtml(item.type||'any');
    const note=item.comment?`<div class="props-help">${escapeHtml(item.comment)}</div>`:'';
    const defaultRow=kind==='inputs'&&item.default!==null&&item.default!==undefined&&item.default!==''
      ?`<div class="prow"><span class="pk">榛樿鍊?</span><span class="pv">${escapeHtml(item.default)}<</span></div>`
      :'';
    return `
      <div class="pgroup pgroup--compact">
        <div class="prow"><span class="pk">${title}<</span>${status}</div>
        <div class="prow"><span class="pk">类型<</span><span class="pv">${type}<</span></div>
        ${defaultRow}
        ${note}
      </div>`;
  }).join('');
}

getPythonBindingSummary = function(node){
  const binding = ensureFlowNodePythonBinding(node);
  return `${binding.portMapping.inputs.length} 输入 / ${binding.portMapping.outputs.length} 输出 / ${binding.portMapping.middleVars.length} 中间变量`;
};

buildPythonPortTooltip = function(meta,kind='input'){
  if(!meta){return '';}
  const parts=[meta.displayName||meta.varName||'端口'];
  if(meta.type){parts.push(`类型: ${meta.type}`);}
  if(kind==='input'&&meta.default!==null&&meta.default!==undefined&&meta.default!==''){
    parts.push(`默认值: ${meta.default}`);
  }
  if(meta.comment){parts.push(meta.comment);}
  return parts.join('\n');
};

buildPythonBindingPortRows = function(items,kind){
  if(!items.length){
    return '<div class="props-help">当前没有可用端口。</div>';
  }
  return items.map(item=>{
    const status=kind==='inputs'
      ?(item.connected?'<span class="pv pv-ok">已连接<</span>':'<span class="pv pv-fault">未连接<</span>')
      :'<span class="pv pv-ok">已暴露<</span>';
    const title=escapeHtml(item.displayName||item.varName||'-');
    const type=escapeHtml(item.type||'any');
    const note=item.comment?`<div class="props-help">${escapeHtml(item.comment)}</div>`:'';
    const defaultRow=kind==='inputs'&&item.default!==null&&item.default!==undefined&&item.default!==''
      ?`<div class="prow"><span class="pk">默认值<</span><span class="pv">${escapeHtml(item.default)}<</span></div>`
      :'';
    return `
      <div class="pgroup pgroup--compact">
        <div class="prow"><span class="pk">${title}<</span>${status}</div>
        <div class="prow"><span class="pk">类型<</span><span class="pv">${type}<</span></div>
        ${defaultRow}
        ${note}
      </div>`;
  }).join('');
};

openPythonBindingDialog = function(nodeId=''){
  const target=getNode(nodeId||S.selBlk);
  if(!target||target.type!=='simulation_block'){
    toast('请先选中一个仿真块以绑定 Python 文件','w');
    return;
  }
  const binding=ensureFlowNodePythonBinding(target);
  window.dispatchEvent(new CustomEvent('gz:open-python-binding',{
    detail:{
      targetNode:{id:target.id,label:target.props.name,props:cloneDefaults(target.props)},
      parsedInterface:binding.bound?cloneDefaults(binding.parsedInterface):null,
      boundSnapshot:binding.bound?cloneDefaults(binding):null
    }
  }));
};

applyPythonBindingToNode = function(nodeId,binding){
  const node=getNode(nodeId);
  if(!node||node.type!=='simulation_block'){return;}
  node.pythonBinding=cloneDefaults(binding);
  applyNodeGeometry(node);
  const removedEdges=pruneInvalidEdgesForNode(node);
  syncAllPythonInputConnectionState();
  markTopologyDirty('system');
  renderModelNodes();
  selectNode(node.id);
  syncConfigTargets(node.id);
  updateUI();
  toast(`已为 ${node.props.name} 绑定 ${binding.fileName}${removedEdges?`，并移除 ${removedEdges} 条无效连线`:''}`,'s');
};

unbindPythonBinding = function(nodeId=''){
  const node=getNode(nodeId||S.selBlk);
  if(!node||node.type!=='simulation_block'){return;}
  const binding=ensureFlowNodePythonBinding(node);
  if(!binding.bound){
    toast('当前仿真块尚未绑定 Python 文件','w');
    return;
  }
  node.pythonBinding=createDefaultPythonBinding();
  applyNodeGeometry(node);
  const removedEdges=pruneInvalidEdgesForNode(node);
  syncAllPythonInputConnectionState();
  markTopologyDirty('system');
  renderModelNodes();
  selectNode(node.id);
  syncConfigTargets(node.id);
  updateUI();
  toast(`已解除 ${node.props.name} 的 Python 绑定${removedEdges?`，并移除 ${removedEdges} 条无效连线`:''}`,'s');
};

getNodePorts = function(node){
  if(!node){return {inputs:[],outputs:[]};}
  if(node.type==='simulation_block'&&isPythonBoundFlowBlock(node)){
    const binding=ensureFlowNodePythonBinding(node);
    return {
      inputs:binding.portMapping.inputs.map((item,index)=>(
        {key:`input-${index}`,label:item.displayName||item.varName||`input_${index}`,kind:'input',side:'left',meta:item}
      )),
      outputs:[
        ...binding.portMapping.outputs.map((item,index)=>(
          {key:`output-${index}`,label:item.displayName||item.varName||`output_${index}`,kind:'output',side:'right',meta:item}
        )),
        ...binding.portMapping.middleVars.map((item,index)=>(
          {key:`middle-${index}`,label:item.displayName||item.varName||`middle_${index}`,kind:'middle',side:'top',meta:item}
        ))
      ]
    };
  }
  if(node.type==='flow_block'){
    return {
      inputs:[{key:'input',label:node.props.inputName||'输入',kind:'input',side:'left'}],
      outputs:[{key:'output',label:node.props.outputName||'输出',kind:'output',side:'right'}]
    };
  }
  return __legacyGetNodePorts(node);
};

createNode = function(type,rawX,rawY){
  const beforeLength=S.modelNodes.length;
  __legacyCreateNode(type,rawX,rawY);
  if(S.modelNodes.length!==beforeLength+1){return;}
  const node=S.modelNodes[S.modelNodes.length-1];
  if(!node){return;}
  if(type==='flow_block'){
    node.pythonBinding=null;
  }
  if(type==='simulation_block'&&!node.pythonBinding){
    node.pythonBinding=createDefaultPythonBinding();
  }
  applyNodeGeometry(node);
  renderModelNodes();
  if(S.selBlk===node.id){
    renderPropertyPanel(node);
  }
  updateUI();
};

renderModelNodes = function(){
  __legacyRenderModelNodes();
  S.modelNodes.forEach(node=>{
    const el=document.getElementById(`b-${node.id}`);
    if(!el){return;}
    if(node.type==='flow_block'){
      el.title='流程块用于定义输入输出名称与数据类型';
      el.addEventListener('click',event=>{
        if(event.detail>=2){
          event.stopImmediatePropagation();
          selectNode(node.id);
        }
      },true);
      el.addEventListener('dblclick',event=>{
        event.stopImmediatePropagation();
      },true);
      const kicker=el.querySelector('.blk-kicker');
      const sub=el.querySelector('.blk-sub');
      if(kicker){kicker.textContent='接口';}
      if(sub){sub.textContent='流程块 · 输入输出类型校验';}
      return;
    }
    if(node.type==='simulation_block'){
      const pythonBound=isPythonBoundFlowBlock(node);
      el.title=pythonBound
        ?`已绑定 Python：${node.pythonBinding?.fileName||'unknown.py'}\n双击可重新绑定`
        :'双击绑定 Python 文件';
      el.addEventListener('dblclick',event=>{
        event.stopPropagation();
        openPythonBindingDialog(node.id);
      });
      const kicker=el.querySelector('.blk-kicker');
      const sub=el.querySelector('.blk-sub');
      if(pythonBound){
        if(kicker){kicker.textContent='Python 已绑定';}
        if(sub){sub.textContent=`仿真块 · ${getPythonBindingSummary(node)}`;}
      }else if(kicker){
        kicker.textContent='仿真';
      }
    }
  });
};

saveSelectedNode = function(){
  const node=getNode(S.selBlk);
  if(!node){return;}
  if(node.type==='signal_source'){
    node.props.name=document.getElementById('prop-name').value.trim()||node.props.name;
    node.props.waveType=document.getElementById('prop-waveType').value;
    node.props.outputFormat=document.getElementById('prop-outputFormat').value;
    node.props.amplitude=document.getElementById('prop-amplitude').value.trim()||node.props.amplitude;
    node.props.frequency=document.getElementById('prop-frequency').value.trim()||node.props.frequency;
    markTopologyDirty(getMutationScopeForType(node.type));
    renderModelNodes();
    renderPropertyPanel(node);
    syncConfigTargets(node.id);
    updateUI();
    return;
  }
  if(node.type==='flow_block'){
    node.props.name=document.getElementById('prop-name').value.trim()||node.props.name;
    node.props.inputName=document.getElementById('prop-inputName').value.trim()||'输入信号';
    node.props.inputFormat=document.getElementById('prop-inputFormat').value;
    node.props.outputName=document.getElementById('prop-outputName').value.trim()||'输出信号';
    node.props.outputFormat=document.getElementById('prop-outputFormat').value;
    markTopologyDirty(getMutationScopeForType(node.type));
    renderModelNodes();
    renderPropertyPanel(node);
    syncConfigTargets(node.id);
    updateUI();
    return;
  }
  if(node.type==='simulation_block'){
    node.props.name=document.getElementById('prop-name').value.trim()||node.props.name;
    if(!isPythonBoundFlowBlock(node)){
      node.props.moduleType=document.getElementById('prop-moduleType').value;
      node.props.inputs=readSimulationInterfaceRows('输入');
      node.props.outputs=readSimulationInterfaceRows('输出');
      node.props.middleVars=readSimulationInterfaceRows('中间量');
      normalizeSimulationInterfaces(node.props);
      node.props.inputs=normalizeSimulationInterfaceList(
        node.props.inputs.slice(0,clampInterfaceCount(document.getElementById('sim-input-count')?.value,SIM_INTERFACE_LIMITS.inputs)),
        '输入',
        node.props.inputs[0]?.type||'标量',
        clampInterfaceCount(document.getElementById('sim-input-count')?.value,SIM_INTERFACE_LIMITS.inputs)
      );
      node.props.outputs=normalizeSimulationInterfaceList(
        node.props.outputs.slice(0,clampInterfaceCount(document.getElementById('sim-output-count')?.value,SIM_INTERFACE_LIMITS.outputs)),
        '输出',
        node.props.outputs[0]?.type||'标量',
        clampInterfaceCount(document.getElementById('sim-output-count')?.value,SIM_INTERFACE_LIMITS.outputs)
      );
      node.props.middleVars=normalizeSimulationInterfaceList(
        node.props.middleVars.slice(0,clampInterfaceCount(document.getElementById('sim-middle-count')?.value,SIM_INTERFACE_LIMITS.middleVars)),
        '中间量',
        node.props.middleVars[0]?.type||'标量',
        clampInterfaceCount(document.getElementById('sim-middle-count')?.value,SIM_INTERFACE_LIMITS.middleVars)
      );
      node.props.ffunctionNote=document.getElementById('prop-ffunctionNote').value.trim()||'当前预留 Python 函数接口，可在后续导入函数定义。';
      applyNodeGeometry(node);
      pruneInvalidEdgesForNode(node);
    }
    markTopologyDirty(getMutationScopeForType(node.type));
    renderModelNodes();
    renderPropertyPanel(node);
    syncConfigTargets(node.id);
    updateUI();
    return;
  }
  return __legacySaveSelectedNode();
};

  if(node?.type==='simulation_block'&&isPythonBoundFlowBlock(node)){
    toast('已绑定 Python 的仿真块不支持手动调整接口数量','w');
    return;
  }
  return __legacySyncSimulationInterfaceCounts();
};

resolveNodeOutputs = function(nodeId,modeKey,time,dt,cache,visiting){
  const node=getNode(nodeId);
  if(!(node&&node.type==='simulation_block'&&isPythonBoundFlowBlock(node))){
    return __legacyResolveNodeOutputs(nodeId,modeKey,time,dt,cache,visiting);
  }
  if(cache.has(nodeId)){return cache.get(nodeId);}
  const bucket=modeKey==='reference'?SIM.reference:SIM.actual;
  const nodeState=ensureNodeRuntime(bucket,node);
  if(visiting.has(nodeId)){
    return {
      outputs:nodeState.lastOutputs?.length?nodeState.lastOutputs:[nodeState.lastOutput||0],
      middleValues:nodeState.lastMiddleValues||[]
    };
  }
  visiting.add(nodeId);
  const inputValues=resolveNodeInputValues(node,modeKey,time,dt,cache,visiting);
  let resolved=executeBoundFlowBlock(node,nodeState,inputValues,modeKey,time,dt);
  resolved.outputs=(resolved.outputs||[0]).map(sanitizeSignalValue);
  resolved.middleValues=(resolved.middleValues||[]).map(sanitizeSignalValue);
  nodeState.lastInputValues=inputValues.slice();
  nodeState.lastInput=inputValues[0]||0;
  nodeState.lastOutputs=resolved.outputs.slice();
  nodeState.lastMiddleValues=resolved.middleValues.slice();
  nodeState.lastOutput=nodeState.lastOutputs[0]||0;
  visiting.delete(nodeId);
  cache.set(nodeId,resolved);
  return resolved;
};

resolveNodeOutput = function(nodeId,outputIndex,modeKey,time,dt,cache,visiting){
  const node=getNode(nodeId);
  if(!(node&&node.type==='simulation_block'&&isPythonBoundFlowBlock(node))){
    return __legacyResolveNodeOutput(nodeId,outputIndex,modeKey,time,dt,cache,visiting);
  }
  const resolved=resolveNodeOutputs(nodeId,modeKey,time,dt,cache,visiting);
  const mainCount=node.pythonBinding?.portMapping?.outputs?.length||0;
  if(outputIndex<mainCount){
    return sanitizeSignalValue(resolved.outputs[outputIndex]??resolved.outputs[0]??0);
  }
  return sanitizeSignalValue(resolved.middleValues[outputIndex-mainCount]??resolved.outputs[0]??0);
};

async function pickJsonFileText(){
  return new Promise((resolve,reject)=>{
    const input=document.createElement('input');
    input.type='file';
    input.accept='.json,application/json';
    input.onchange=async ()=>{
      const file=input.files&&input.files[0];
      if(!file){
        resolve(null);
        return;
      }
      try{
        resolve(await file.text());
      }catch(error){
        reject(error);
      }
    };
    input.click();
  });
}

  link.href=url;
  link.download=`${safeModelId}.json`;
  document.body.appendChild(link);
  link.click();
  link.remove();
  setTimeout(()=>URL.revokeObjectURL(url),0);
}

buildNodeSubtitle = function(node){
  const props=node.props||{};
  const faultNote=node.injectedFault?' · 电气故障':'';
  switch(node.type){

      return `${props.waveType||'正弦'} · 幅值 ${props.amplitude} · ${props.outputFormat||'标量'}`;
    case 'flow_block':
      return `${props.inputName||'输入信号'} / ${props.inputFormat||'标量'} → ${props.outputName||'输出信号'} / ${props.outputFormat||'标量'}${faultNote}`;
    case 'simulation_block':
      normalizeSimulationInterfaces(props);
      return `${props.inputs.length}输入 · ${props.outputs.length}输出 · ${props.middleVars.length}中间变量 · ${props.moduleType||'一阶函数'}${faultNote}`;
    case 'fault_bias':
    case 'fault_noise':
    case 'fault_stuck':
    case 'fault_drift':
      return `${props.faultType||'故障'} · ${props.layer||'物理层'}`;
    case 'instrument_scope':
      return `${props.instrumentType||'示波器'} · 双击查看波形`;
    case 'instrument_spectrum':
    case 'instrument_logger':
      return `${props.instrumentType||'仪器'} · ${props.sampleRate||'1000 Hz'}`;
    case 'link_normal':
    case 'link_can':
      return `${props.sourcePort||'源端口'} → ${props.targetPort||'目标端口'}`;
    default:
      return '';
  }
};

computeSourceSignal = function(node,time){
  const amplitude=parseScalar(node.props.amplitude,1);
  const frequency=parseScalar(node.props.frequency,1);
  if(node.props.waveType==='阶跃'){
    const stepTime=Math.max(0.2,frequency);
    return time>=stepTime?amplitude:0;
  }
  if(node.props.waveType==='常值'){
    return amplitude;
  }
  return amplitude*Math.sin(2*Math.PI*frequency*time);
};

applyElectricalFault = function(node,inputValue,outputValue,time){
  if(!node.injectedFault){return outputValue;}
  const name=node.injectedFault.name||'';
  if(name.includes('过流')){
    return Math.abs(inputValue)>1.2?0:outputValue*0.78;
  }
  if(name.includes('短路')||name.includes('断路')||name.includes('开路')){
    return 0;
  }
  if(name.includes('漂移')){
    return outputValue+Math.min(0.4,time*0.08);
  }
  return outputValue;
};

getScopeSignalName = function(scopeNode){
  const incoming=getIncomingEdgesForNode(scopeNode.id)[0];
  if(!incoming){return '未接入信号';}
  const sourceNode=getNode(incoming.sourceNodeId);
  if(!sourceNode){return '上游信号';}
  const outputPort=getNodePorts(sourceNode).outputs[incoming.sourcePortIndex||0];
  if(outputPort){
    return `${sourceNode.props?.name||'上游信号'} · ${outputPort.label}`;
  }
  return sourceNode.props?.name||'上游信号';
};

  if(!S.sysLoaded){toast('请先导入系统模型','w');return false;}
  if(!S.modelNodes.some(node=>node.type==='signal_source')){
    toast('请至少放置一个信号源','w');
    return false;
  }
  if(getScopeNodes().length===0){
    toast('请至少放置一个示波器用于观察信号','w');

  if(getConnectedScopeNodes().length===0){
    toast('请至少让一个示波器接入信号链路，才能查看波形','w');
    return false;
  }
  const conflicts=findInputCouplingConflicts();

    toast('检测到同一输入端口存在多条连线，请先删除多余连线以避免信号耦合','w');
    return false;
  }
  return true;
};

simStep = function(){
  if(!SIM.inited){
    simInit();
    if(!SIM.inited){return;}
  }
  if(SIM.time>=SIM.duration){
    toast('当前仿真已结束，请重新初始化后再步进','w');
    return;
  }
  const completed=runSimulationTick();
  updateSimbarEnabled();
  if(completed){
    simStop('completed');
  }else{
    toast(`步进完成 · t = ${SIM.time.toFixed(2)}s`,'s');
  }
};

simPause = function(){
  if(!SIM.running){return;}
  SIM.paused=!SIM.paused;
  if(SIM.paused){
    clearSimLoop();
    toast('仿真已暂停','w');
  }else{
    startSimulationLoop();
    toast('仿真继续运行','s');
  }
  updateSimbarEnabled();
  updateUI();
};

function buildNamedOptions(options, selectedValue){
  return options.map(option=>{
    const value=typeof option==='string'?option:option.value;
    const label=typeof option==='string'?option:(option.label||option.value);
    return `<option value="${escapeHtml(value)}"${String(value)===String(selectedValue)?' selected':''}>${escapeHtml(label)}</option>`;
  }).join('');
}

function getSimulationMiddleVarDefinitions(node){
  if(!node||node.type!=='simulation_block'){
    return [];
  }
  if(isPythonBoundFlowBlock(node)){
    return (ensureFlowNodePythonBinding(node).portMapping.middleVars||[]).map((item,index)=>({
      name:item.displayName||item.varName||`middle_${index}`,
      type:item.type||'标量',
      varName:item.varName||`middle_${index}`,
      displayName:item.displayName||item.varName||`middle_${index}`
    }));
  }
  normalizeSimulationInterfaces(node.props);
  return (node.props.middleVars||[]).map((item,index)=>({
    name:item.name||`中间量 ${index+1}`,
    type:item.type||'标量',
    varName:item.name||`中间量_${index+1}`,
    displayName:item.name||`中间量 ${index+1}`
  }));
}

function getMiddleAssignmentResolvedTarget(node){
  const targetNode=getNode(node?.props?.targetNodeId||'');
  if(!targetNode||targetNode.type!=='simulation_block'){
    return null;
  }
  const defs=getSimulationMiddleVarDefinitions(targetNode);
  const targetVar=defs.find(item=>item.name===node.props.targetVarName)||defs[0]||null;
  if(!targetVar){
    return null;
  }
  return {targetNode,targetVar};
}

function clearAssignedMiddleValues(bucket){
  if(!bucket?.nodeState){
    return;
  }
  Object.values(bucket.nodeState).forEach(nodeState=>{
    if(nodeState&&nodeState.assignedMiddleValues){
      nodeState.assignedMiddleValues={};
    }
  });
}

function readMiddleAssignmentValue(node,modeKey,time,dt,cache,visiting){
  if(node.props.assignMode==='input'){
    const inputValues=resolveNodeInputValues(node,modeKey,time,dt,cache,visiting);
    return sanitizeSignalValue(inputValues[0]??0);
  }
  return sanitizeSignalValue(parseScalar(node.props.constantValue,0));
}

function applyMiddleAssignmentsForMode(modeKey,time,dt,cache,visiting){
  const bucket=modeKey==='reference'?SIM.reference:SIM.actual;
  S.modelNodes.filter(node=>node.type==='middle_var_assign').forEach(node=>{
    const targetInfo=getMiddleAssignmentResolvedTarget(node);
    const nodeState=ensureNodeRuntime(bucket,node);
    if(!targetInfo){
      nodeState.lastAssignedValue=0;
      return;
    }
    const value=readMiddleAssignmentValue(node,modeKey,time,dt,cache,visiting);
    nodeState.lastAssignedValue=value;
    const targetState=ensureNodeRuntime(bucket,targetInfo.targetNode);
    if(!targetState.assignedMiddleValues){
      targetState.assignedMiddleValues={};
    }
    targetState.assignedMiddleValues[targetInfo.targetVar.name]=value;
  });
}

function primeSimulationNodeStateFromAssignments(node,nodeState){
  const assigned=nodeState?.assignedMiddleValues||{};
  const middleDefs=getSimulationMiddleVarDefinitions(node);
  const firstVar=middleDefs[0];
  if(!firstVar||!Object.prototype.hasOwnProperty.call(assigned,firstVar.name)){
    return;
  }
  const seed=sanitizeSignalValue(assigned[firstVar.name]);
  if(node.props.moduleType==='积分'){
    nodeState.integrator=seed;
  }else if(node.props.moduleType==='微分'){
    nodeState.prevAggregateInput=seed;
  }else{
    nodeState.firstOrder=seed;
  }
}

function applyAssignedMiddleValues(node,nodeState,resolved){
  const assigned=nodeState?.assignedMiddleValues||{};
  const middleDefs=getSimulationMiddleVarDefinitions(node);
  if(!resolved.middleValues){
    resolved.middleValues=[];
  }
  const firstAssignedName=middleDefs[0]?.name;
  const firstAssignedValue=firstAssignedName&&Object.prototype.hasOwnProperty.call(assigned,firstAssignedName)
    ?sanitizeSignalValue(assigned[firstAssignedName])
    :undefined;
  middleDefs.forEach((item,index)=>{
    if(Object.prototype.hasOwnProperty.call(assigned,item.name)){
      resolved.middleValues[index]=sanitizeSignalValue(assigned[item.name]);
    }
  });
  if(firstAssignedValue!==undefined&&resolved.outputs?.length){
    resolved.outputs[0]=firstAssignedValue;
    if(resolved.outputs.length>1){
      resolved.outputs[1]=firstAssignedValue;
    }
  }
  return resolved;
}

getNodePorts = function(node){
  if(!node){
    return {inputs:[],outputs:[]};
  }
  if(node.type==='signal_source'){
    return {
      inputs:[],
      outputs:[{key:'signal',label:node.props.outputFormat||'输出',kind:'output',side:'right'}]
    };
  }
  if(node.type==='flow_block'){
    return {
      inputs:[{key:'input',label:node.props.inputName||'输入',kind:'input',side:'left'}],
      outputs:[{key:'output',label:node.props.outputName||'输出',kind:'output',side:'right'}]
    };
  }
  if(node.type==='simulation_block'){
    if(isPythonBoundFlowBlock(node)){
      const binding=ensureFlowNodePythonBinding(node);
      return {
        inputs:binding.portMapping.inputs.map((item,index)=>({
          key:`input-${index}`,
          label:item.displayName||item.varName||`input_${index}`,
          kind:'input',
          side:'left',
          meta:item
        })),
        outputs:[
          ...binding.portMapping.outputs.map((item,index)=>({
            key:`output-${index}`,
            label:item.displayName||item.varName||`output_${index}`,
            kind:'output',
            side:'right',
            meta:item
          })),
          ...binding.portMapping.middleVars.map((item,index)=>({
            key:`middle-${index}`,
            label:item.displayName||item.varName||`middle_${index}`,
            kind:'middle',
            side:'top',
            meta:item
          }))
        ]
      };
    }
    const groups=getSimulationPortSets(node.props);
    return {
      inputs:groups.inputs.map((item,index)=>({
        key:`input-${index}`,
        label:`输入 ${index+1} · ${item.name} / ${item.type}`,
        kind:'input',
        side:'left'
      })),
      outputs:getSimulationFlatOutputs(node.props).map((item,index)=>({
        key:`${item.kind}-${item.index}`,
        label:`${item.kind==='middle'?'中间量':'输出'} ${item.index+1} · ${item.name} / ${item.type}`,
        kind:item.kind,
        side:item.side,
        portIndex:index
      }))
    };
  }
  if(node.type==='middle_var_assign'){
    return {
      inputs:[{key:'assign-input',label:'赋值输入',kind:'input',side:'left'}],
      outputs:[]
    };
  }
  if(node.type.startsWith('fault_')){
    return {
      inputs:[{key:'input',label:'故障输入',kind:'input',side:'left'}],
      outputs:[{key:'output',label:'故障输出',kind:'output',side:'right'}]
    };
  }
  if(node.type.startsWith('instrument_')){
    return {
      inputs:[{key:'input',label:'观测输入',kind:'input',side:'left'}],
      outputs:[]
    };
  }
  return {inputs:[],outputs:[]};
};

buildNodeSubtitle = function(node){
  const props=node.props||{};
  const faultNote=node.injectedFault?' · 电气故障':'';
  if(node.type==='middle_var_assign'){
    const targetInfo=getMiddleAssignmentResolvedTarget(node);
    const targetName=targetInfo?.targetNode?.props?.name||'未选择目标';
    const varName=targetInfo?.targetVar?.name||'未选择变量';
    const modeLabel=props.assignMode==='input'?'输入覆盖':'常值覆盖';
    return `${targetName} · ${varName} · ${modeLabel}`;
  }
  switch(node.type){
    case 'signal_source':
      return `${props.waveType||'正弦'} · 幅值 ${props.amplitude} · ${props.outputFormat||'标量'}`;
    case 'flow_block':
      return `${props.inputName||'输入'} → ${props.outputName||'输出'} · ${props.outputFormat||'标量'}${faultNote}`;
    case 'simulation_block':
      normalizeSimulationInterfaces(props);
      return `${props.inputs.length}输入 · ${props.outputs.length}输出 · ${props.middleVars.length}中间变量 · ${props.moduleType||'一阶函数'}${faultNote}`;
    case 'instrument_scope':
      return `${props.instrumentType||'示波器'} · 双击查看波形`;
    case 'instrument_spectrum':
    case 'instrument_logger':
      return `${props.instrumentType||COMPONENT_LIBRARY[node.type].label} · 采样率 ${props.sampleRate||'未设置'}`;
    default:
      return __legacyBuildNodeSubtitleClean(node);
  }
};

createNode = function(type,rawX,rawY){
  if(type!=='middle_var_assign'){
    return __legacyCreateNodeClean(type,rawX,rawY);
  }
  const meta=COMPONENT_LIBRARY[type];
  if(!meta){return;}
  const props=cloneDefaults(meta.defaults);
  props.name=`中间变量赋值块 ${getNodeTypeCount(type)}`;
  const point={
    x:clamp(Math.round(rawX-meta.width/2),CANVAS_STAGE.nodePadding,CANVAS_STAGE.width-meta.width-CANVAS_STAGE.nodePadding),
    y:clamp(Math.round(rawY-meta.height/2),CANVAS_STAGE.nodeMinY,CANVAS_STAGE.height-meta.height-CANVAS_STAGE.nodePadding)
  };
  S.modelNodes.push({
    id:`node-${++S.nodeSeq}`,
    type,
    x:point.x,
    y:point.y,
    w:meta.width,
    h:meta.height,
    props,
    pythonBinding:null
  });
  markTopologyDirty('system');
  const node=S.modelNodes[S.modelNodes.length-1];
  renderModelNodes();
  syncConfigTargets(node.id);
  selectNode(node.id);
  updateUI();
  toast(`${meta.label} 已加入画布`,'s');
};

resolveNodeOutputs = function(nodeId,modeKey,time,dt,cache,visiting){
  const node=getNode(nodeId);
  if(node?.type==='middle_var_assign'){
    if(cache.has(nodeId)){
      return cache.get(nodeId);
    }
    const bucket=modeKey==='reference'?SIM.reference:SIM.actual;
    const nodeState=ensureNodeRuntime(bucket,node);
    const value=readMiddleAssignmentValue(node,modeKey,time,dt,cache,visiting);
    nodeState.lastAssignedValue=value;
    const resolved={outputs:[value],middleValues:[]};
    cache.set(nodeId,resolved);
    return resolved;
  }
  const resolved=__legacyResolveNodeOutputsClean(nodeId,modeKey,time,dt,cache,visiting);
  if(node?.type==='simulation_block'&&isPythonBoundFlowBlock(node)){
    const bucket=modeKey==='reference'?SIM.reference:SIM.actual;
    return applyAssignedMiddleValues(node,ensureNodeRuntime(bucket,node),resolved);
  }
  return resolved;
};

function ensureScopeChannelBucket(bucket,scopeId){
  const current=bucket.scopeSamples[scopeId];
  if(!current||Array.isArray(current)){
    bucket.scopeSamples[scopeId]={ch1:[],ch2:[]};
  }else{
    current.ch1=current.ch1||[];
    current.ch2=current.ch2||[];
  }
  return bucket.scopeSamples[scopeId];
}

function getScopeChannelKey(portIndex=0){
  return Number(portIndex)===1?'ch2':'ch1';
}

function getSampleValue(sample){
  if(sample&&Number.isFinite(sample.actual)){return sample.actual;}
  if(sample&&Number.isFinite(sample.v)){return sample.v;}
  return 0;
}

function formatScopeMetric(value){
  return Number.isFinite(value)?value.toFixed(2):'--';
}

function summarizeScopeChannel(samples){
  if(!samples?.length){
    return {connected:false,current:'--',min:'--',max:'--',pp:'--'};
  }
  const values=samples.map(getSampleValue);
  const min=Math.min(...values);
  const max=Math.max(...values);
  return {
    connected:true,
    current:formatScopeMetric(values.at(-1)),
    min:formatScopeMetric(min),
    max:formatScopeMetric(max),
    pp:formatScopeMetric(max-min)
  };
}

function getScopeChannelSignalName(scopeNode,channelKey){
  const targetPortIndex=channelKey==='ch2'?1:0;
  const incoming=getIncomingEdgesForNode(scopeNode.id).find(edge=>(edge.targetPortIndex||0)===targetPortIndex);
  if(!incoming){return '未连接';}
  const sourceNode=getNode(incoming.sourceNodeId);
  if(!sourceNode){return '未知信号';}
  const outputPort=getNodePorts(sourceNode).outputs[incoming.sourcePortIndex||0];
  return outputPort
    ?`${sourceNode.props?.name||'信号'} · ${outputPort.label}`
    :(sourceNode.props?.name||'信号');
}

function prepareScopeCanvas(canvas){
  const ratio=window.devicePixelRatio||1;
  const rect=canvas.getBoundingClientRect();
  const width=Math.max(Math.round(rect.width||canvas.clientWidth||300),1);
  const height=Math.max(Math.round(rect.height||canvas.clientHeight||150),1);
  canvas.width=width*ratio;
  canvas.height=height*ratio;
  const ctx=canvas.getContext('2d');
  if(ctx?.setTransform){
    ctx.setTransform(ratio,0,0,ratio,0,0);
  }
  return {ctx,width,height};
}

function drawScopeTrace(ctx,series,color,getX,getY){
  if(!ctx||!series.length){return;}
  ctx.strokeStyle=color;
  ctx.lineWidth=1.5;
  ctx.beginPath();
  series.forEach((sample,index)=>{
    const x=getX(sample.t,index,series.length);
    const y=getY(getSampleValue(sample));
    if(index===0){ctx.moveTo(x,y);}else{ctx.lineTo(x,y);}
  });
  ctx.stroke();
}

function drawScopeWindow(scopeId){
  const scopeEl=document.querySelector(`.scope-window[data-scope-id="${scopeId}"]`);
  if(!scopeEl){return;}
  const canvas=scopeEl.querySelector('[data-scope-role="wave-canvas"]');
  if(!canvas){return;}
  const {ctx,width,height}=prepareScopeCanvas(canvas);
  if(!ctx){return;}
  const scopeWindow=ensureScopeWindow(scopeId);
  const samples=ensureScopeChannelBucket(SIM.actual,scopeId);
  const ch1Series=samples.ch1.slice(-240);
  const ch2Series=samples.ch2.slice(-240);
  const visibleSeries=scopeWindow.mode==='ch1'
    ?ch1Series
    :scopeWindow.mode==='ch2'
      ?ch2Series
      :ch1Series.concat(ch2Series);
  if(!visibleSeries.length){
    drawScopeEmpty(ctx,width,height,'连接 CH1 或 CH2 后开始仿真');
    return;
  }
  drawScopeGrid(ctx,width,height);
  const values=visibleSeries.map(getSampleValue);
  let minValue=Math.min(...values);
  let maxValue=Math.max(...values);
  if(Math.abs(maxValue-minValue)<1e-6){
    maxValue+=1;
    minValue-=1;
  }else{
    const padding=(maxValue-minValue)*0.18;
    maxValue+=padding;
    minValue-=padding;
  }
  const startTime=Math.min(...visibleSeries.map(sample=>sample.t||0));
  const endTime=Math.max(...visibleSeries.map(sample=>sample.t||0),startTime+SIM.stepSize);
  const span=Math.max(endTime-startTime,SIM.stepSize,0.001);
  const getX=time=>18+((time-startTime)/span)*(width-36);
  const getY=value=>14+(maxValue-value)/(maxValue-minValue)*(height-28);
  const zeroY=getY(0);
  if(zeroY>=10&&zeroY<=height-10){
    ctx.strokeStyle='#c2d6ea';
    ctx.lineWidth=1;
    ctx.beginPath();
    ctx.moveTo(0,zeroY);
    ctx.lineTo(width,zeroY);
    ctx.stroke();
  }
  if(scopeWindow.mode!=='ch2'){
    drawScopeTrace(ctx,ch1Series,'#1d6fbf',getX,getY);
  }
  if(scopeWindow.mode!=='ch1'){
    drawScopeTrace(ctx,ch2Series,'#16a34a',getX,getY);
  }
  ctx.fillStyle='#7a9ab8';
  ctx.font='10px JetBrains Mono,monospace';
  ctx.fillText(`${startTime.toFixed(1)}s`,6,height-6);
  ctx.fillText(`${endTime.toFixed(1)}s`,Math.max(width-44,6),height-6);
}

function startScopeWindowDrag(event,scopeId){
  const scopeWindow=ensureScopeWindow(scopeId);
  focusScopeWindow(scopeId);
  const startX=event.clientX;
  const startY=event.clientY;
  const originX=scopeWindow.x;
  const originY=scopeWindow.y;
  const handleMove=moveEvent=>{
    scopeWindow.x=Math.round(originX+(moveEvent.clientX-startX));
    scopeWindow.y=Math.round(originY+(moveEvent.clientY-startY));
    renderScopeWindows();
  };
  const handleUp=()=>{
    window.removeEventListener('pointermove',handleMove);
    window.removeEventListener('pointerup',handleUp);
    window.removeEventListener('mousemove',handleMove);
    window.removeEventListener('mouseup',handleUp);
    document.removeEventListener('pointermove',handleMove);
    document.removeEventListener('pointerup',handleUp);
    document.removeEventListener('mousemove',handleMove);
    document.removeEventListener('mouseup',handleUp);
  };
  window.addEventListener('pointermove',handleMove);
  window.addEventListener('pointerup',handleUp);
  window.addEventListener('mousemove',handleMove);
  window.addEventListener('mouseup',handleUp);
  document.addEventListener('pointermove',handleMove);
  document.addEventListener('pointerup',handleUp);
  document.addEventListener('mousemove',handleMove);
  document.addEventListener('mouseup',handleUp);
  if(event.stopPropagation){event.stopPropagation();}
  if(event.preventDefault){event.preventDefault();}
}

  window.__GZ_SCOPE_WINDOW_DRAG_BOUND__=true;
  const runtimeListeners=getRuntimeListenerStore();
  runtimeListeners.scopeWindowDragPointerMove=event=>{
    if(!SCOPE_WINDOW_DRAG.active){return;}
    const scopeWindow=SIM.scopeWindows?.[SCOPE_WINDOW_DRAG.scopeId];
    if(!scopeWindow){return;}
    scopeWindow.x=Math.round(SCOPE_WINDOW_DRAG.originX+(event.clientX-SCOPE_WINDOW_DRAG.startX));
    scopeWindow.y=Math.round(SCOPE_WINDOW_DRAG.originY+(event.clientY-SCOPE_WINDOW_DRAG.startY));
    renderScopeWindows();
  };
  runtimeListeners.scopeWindowDragPointerUp=event=>{
    if(!SCOPE_WINDOW_DRAG.active){return;}
    SCOPE_WINDOW_DRAG.active=false;
    SCOPE_WINDOW_DRAG.pointerId=null;
    SCOPE_WINDOW_DRAG.scopeId='';
  };
  window.addEventListener('pointermove',runtimeListeners.scopeWindowDragPointerMove);
  window.addEventListener('pointerup',runtimeListeners.scopeWindowDragPointerUp);
}

        <section class="scope-window" data-scope-id="${scopeWindow.scopeId}" style="left:${scopeWindow.x}px;top:${scopeWindow.y}px;z-index:${scopeWindow.z}">
          <header class="scope-window__header" data-scope-drag-handle="${scopeWindow.scopeId}">
            <div class="scope-window__identity">
              <div class="scope-window__title">${escapeHtml(node?.props?.name||node?.name||'示波器')}</div>
              <div class="scope-window__signals">
                <span class="scope-window__tag" data-scope-tag="ch1" data-connected="${ch1.connected}">CH1: ${escapeHtml(ch1Signal)}</span>
                <span class="scope-window__tag" data-scope-tag="ch2" data-connected="${ch2.connected}">CH2: ${escapeHtml(ch2Signal)}</span>
              </div>
            </div>
            <div class="scope-window__toolbar">
              <button type="button" class="scope-window__mode${mode==='overlay'?' is-active':''}" data-scope-mode="overlay" data-scope-id="${scopeWindow.scopeId}">叠加</button>
              <button type="button" class="scope-window__mode${mode==='ch1'?' is-active':''}" data-scope-mode="ch1" data-scope-id="${scopeWindow.scopeId}">CH1</button>
              <button type="button" class="scope-window__mode${mode==='ch2'?' is-active':''}" data-scope-mode="ch2" data-scope-id="${scopeWindow.scopeId}">CH2</button>
              <button type="button" class="scope-window__close" data-scope-close="${scopeWindow.scopeId}">关闭 ×</button>
            </div>
          </header>
          <div class="scope-window__plot">
            <canvas data-scope-role="wave-canvas"></canvas>
          </div>
          <div class="scope-window__stats scope-window__stats--strip">
            <article class="scope-readout scope-readout--strip" data-scope-channel="ch1" data-connected="${ch1.connected}">
              <div class="scope-readout__title">CH1</div>
              <div class="scope-readout__metric"><span>当前</span><strong data-field="current">${ch1.current}</strong></div>
              <div class="scope-readout__metric"><span>最小</span><strong data-field="min">${ch1.min}</strong></div>
              <div class="scope-readout__metric"><span>最大</span><strong data-field="max">${ch1.max}</strong></div>
              <div class="scope-readout__metric"><span>峰峰</span><strong data-field="pp">${ch1.pp}</strong></div>
            </article>
            <article class="scope-readout scope-readout--strip" data-scope-channel="ch2" data-connected="${ch2.connected}">
              <div class="scope-readout__title">CH2</div>
              <div class="scope-readout__metric"><span>当前</span><strong data-field="current">${ch2.current}</strong></div>
              <div class="scope-readout__metric"><span>最小</span><strong data-field="min">${ch2.min}</strong></div>
              <div class="scope-readout__metric"><span>最大</span><strong data-field="max">${ch2.max}</strong></div>
              <div class="scope-readout__metric"><span>峰峰</span><strong data-field="pp">${ch2.pp}</strong></div>
            </article>
          </div>
        </section>`;
    })
    .join('');
  host.querySelectorAll('.scope-window').forEach(el=>{
    el.addEventListener('pointerdown',event=>{
      if(event.target.closest('[data-scope-close],[data-scope-mode],[data-scope-drag-handle]')){return;}
      const scopeWindow=focusScopeWindow(el.dataset.scopeId||'');
      el.style.zIndex=scopeWindow.z;
    });
  });
  host.querySelectorAll('[data-scope-drag-handle]').forEach(header=>{
    header.addEventListener('pointerdown',event=>{
      startScopeWindowDrag(event,header.dataset.scopeDragHandle||'');
    });
  });
  host.querySelectorAll('[data-scope-close]').forEach(btn=>{
    btn.addEventListener('click',event=>{
      event.stopPropagation();
      closeScopeWindow(btn.dataset.scopeClose||'');
    });
  });
  host.querySelectorAll('[data-scope-mode]').forEach(btn=>{
    btn.addEventListener('click',event=>{
      event.stopPropagation();
      setScopeDisplayMode(btn.dataset.scopeId||'',btn.dataset.scopeMode||'overlay');
    });
  });
  listOpenScopeWindows().forEach(scopeWindow=>drawScopeWindow(scopeWindow.scopeId));
};

  renderScopeWindows();
};

runSimulationTick = function(){
  const completed=__dualScopeRunSimulationTick();
  if(listOpenScopeWindows().length){
    renderScopeWindows();
  }
  return completed;
};

openScope = function(scopeId=''){
  ensureScopeSelection(scopeId||S.selBlk);
  if(!SIM.activeScopeId){
    toast('当前没有可打开的示波器模块','w');
    return;
  }
  focusScopeWindow(SIM.activeScopeId);
  renderScopeWindows();
};

focusScopeLatest = function(){
  if(!SIM.activeScopeId){return;}
  focusScopeWindow(SIM.activeScopeId);
  renderScopeWindows();
};

togCmp = function(){
  S.cmpMode=!S.cmpMode;
  const scopeId=SIM.activeScopeId||getScopeNodes()[0]?.id||'';
  if(scopeId){
    setScopeDisplayMode(scopeId,S.cmpMode?'overlay':'ch1');
  }
};
